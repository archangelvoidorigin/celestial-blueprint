## Context

The observer script (T5.2.01) runs as a standalone process, but agents and the frontend also need to trigger observation on demand. `POST /api/blueprints/{id}/observe` runs the same drift-detection logic inline (without subprocess) and returns the list of findings created. The logic is identical to `observe_blueprint()` in the script but uses the internal store directly rather than HTTP. The endpoint creates assumption DB records directly rather than making HTTP calls to itself.

## Objective

Create `app/api/observe.py` with a `POST /api/blueprints/{id}/observe` endpoint that runs inline drift detection and returns findings.

## Instructions

1. Create `backend/app/api/observe.py` with exactly:
   ```python
   """On-demand observer: detect blueprint drift vs. project state."""
   from __future__ import annotations

   import subprocess
   from datetime import datetime, timezone

   from fastapi import APIRouter, HTTPException

   from app.services import store

   router = APIRouter(prefix="/api/blueprints", tags=["observer"])


   def _latest_git_commit_ts() -> datetime | None:
       try:
           out = subprocess.check_output(
               ["git", "log", "-1", "--format=%ct"],
               stderr=subprocess.DEVNULL, text=True,
           ).strip()
           return datetime.fromtimestamp(int(out), tz=timezone.utc) if out else None
       except Exception:
           return None


   def _staleness_days(updated_at: str, now: datetime) -> float:
       try:
           dt = datetime.fromisoformat(updated_at.replace("Z", "+00:00"))
           if dt.tzinfo is None:
               dt = dt.replace(tzinfo=timezone.utc)
           return (now - dt).total_seconds() / 86400
       except Exception:
           return 0.0


   @router.post("/{bid}/observe")
   def observe_blueprint(bid: str, stale_days: float = 3.0):
       bp = store.get_blueprint(bid)
       if bp is None:
           raise HTTPException(status_code=404, detail="blueprint not found")
       fm = bp.get("frontmatter", {})
       status = fm.get("status", "")
       if status not in ("draft", "active", "in-progress"):
           return {"blueprint_id": bid, "status": status, "findings": [], "message": "blueprint not active — skipped"}

       now = datetime.now(tz=timezone.utc)
       commit_ts = _latest_git_commit_ts()
       updated = fm.get("modified", fm.get("date_created", ""))
       days = _staleness_days(updated, now) if updated else 0.0
       findings = []

       if days >= stale_days and commit_ts is not None:
           assumption_id = f"obs-drift-{now.strftime('%Y%m%d%H%M%S')}"
           body = f"Reality drift — blueprint not updated in {days:.0f} days but project has recent commits"
           store.add_assumption(bid, {
               "id": assumption_id, "status": "unvalidated", "risk": "high",
               "bp_ref": "", "body": body,
           })
           findings.append({"type": "drift", "assumption_id": assumption_id, "message": body})

       return {"blueprint_id": bid, "stale_days_checked": stale_days, "days_since_update": round(days, 1), "findings": findings}
   ```
