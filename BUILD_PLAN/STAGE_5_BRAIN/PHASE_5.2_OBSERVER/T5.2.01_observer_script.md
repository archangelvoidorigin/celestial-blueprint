## Context

The Observer checks whether active blueprints are drifting from reality. It queries the Celestial API for blueprints, computes a "staleness score" based on the blueprint's `modified` frontmatter date (the canonical field set by `default_frontmatter`/`patch_blueprint` in T1.3.04 — there is no `updated_at` key in frontmatter) vs. the git log's latest commit in the project root, and for each stale blueprint logs an assumption entry (`"Reality drift detected — blueprint not updated in N days"`) via the API. The script is a standalone CLI tool run by Hermes or manually; it uses `httpx` from the backend venv and reads `CELESTIAL_API_URL` and `CELESTIAL_AUTH_TOKEN` from the environment.

## Objective

   ```python
   #!/usr/bin/env python3
   """
   Celestial Blueprint Observer

   Detects active blueprints that have not been updated while the project has
   had git commits, and logs a drift assumption for each stale one.

   Usage:
     python3 tools/observer.py                  # check all active blueprints
     python3 tools/observer.py --id <bp-id>     # check one blueprint
     python3 tools/observer.py --dry-run        # print findings, no writes
     python3 tools/observer.py --stale-days 7   # threshold (default: 3)
   """
   from __future__ import annotations

   import argparse
   import os
   import subprocess
   from datetime import datetime, timezone

   import httpx

   API_URL: str = os.environ.get("CELESTIAL_API_URL", "http://localhost:8787")
   TOKEN: str = os.environ.get("CELESTIAL_AUTH_TOKEN", "")
   HEADERS: dict[str, str] = {"Authorization": f"Bearer {TOKEN}"} if TOKEN else {}


   def latest_git_commit_ts() -> datetime | None:
       """Return UTC datetime of the most recent git commit in the repo root."""
       try:
           out = subprocess.check_output(
               ["git", "log", "-1", "--format=%ct"],
               stderr=subprocess.DEVNULL,
               text=True,
           ).strip()
           return datetime.fromtimestamp(int(out), tz=timezone.utc) if out else None
       except Exception:
           return None


   def staleness_days(updated_at: str, now: datetime) -> float:
       """Return how many days ago updated_at was. updated_at is ISO-8601 or date."""
       try:
           dt = datetime.fromisoformat(updated_at.replace("Z", "+00:00"))
           if dt.tzinfo is None:
               dt = dt.replace(tzinfo=timezone.utc)
           return (now - dt).total_seconds() / 86400
       except Exception:
           return 0.0


   def observe_blueprint(bid: str, dry_run: bool, stale_days: float, now: datetime, commit_ts: datetime | None) -> list[str]:
       """Check one blueprint; return list of finding strings."""
       r = httpx.get(f"{API_URL}/api/blueprints/{bid}", headers=HEADERS, timeout=15)
       if r.status_code != 200:
           return [f"  SKIP {bid}: {r.status_code}"]
       bp = r.json()
       fm = bp.get("frontmatter", {})
       status = fm.get("status", "")
       if status not in ("draft", "active", "in-progress"):
           return []
       updated = fm.get("modified", fm.get("date_created", ""))
       days = staleness_days(updated, now) if updated else 0.0
       findings = []
       if days >= stale_days and commit_ts is not None:
           msg = f"Reality drift — blueprint not updated in {days:.0f} days but project has recent commits"
           findings.append(f"  DRIFT {bid}: {msg}")
           if not dry_run:
               httpx.post(
                   f"{API_URL}/api/blueprints/{bid}/assumptions",
                   json={"id": f"obs-drift-{now.strftime('%Y%m%d')}", "status": "unvalidated",
                         "risk": "high", "body": msg},
                   headers=HEADERS,
                   timeout=15,
               )
       return findings


   def main() -> None:
       parser = argparse.ArgumentParser(description="Celestial Blueprint Observer")
       parser.add_argument("--id", dest="bp_id", default="", help="Check one blueprint ID")
       parser.add_argument("--dry-run", action="store_true", help="Print findings, no writes")
       parser.add_argument("--stale-days", type=float, default=3.0, help="Drift threshold in days (default 3)")
       args = parser.parse_args()

       now = datetime.now(tz=timezone.utc)
       commit_ts = latest_git_commit_ts()

       if args.bp_id:
           bps = [{"id": args.bp_id}]
       else:
           resp = httpx.get(f"{API_URL}/api/blueprints", headers=HEADERS, timeout=15)
           resp.raise_for_status()
           bps = resp.json()

       total_drift = 0
       for bp in bps:
           findings = observe_blueprint(bp.get("id", bp.get("frontmatter", {}).get("id", "")),
                                        args.dry_run, args.stale_days, now, commit_ts)
           for f in findings:
               print(f)
               total_drift += 1

       mode = "DRY-RUN " if args.dry_run else ""
       print(f"Observer {mode}complete: {len(bps)} blueprint(s) checked, {total_drift} drift(s) found")


   if __name__ == "__main__":
       main()
   ```
