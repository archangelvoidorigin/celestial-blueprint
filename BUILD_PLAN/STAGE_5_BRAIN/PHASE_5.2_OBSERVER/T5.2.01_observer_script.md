## Context

The Observer checks whether active blueprints are drifting from reality. It performs **two** checks per blueprint:

1. **Staleness (timestamp drift)** — compares the blueprint's `modified` frontmatter date (the canonical field set by `default_frontmatter`/`patch_blueprint` in T1.3.04 — there is no `updated_at` key) against the latest git commit in the project root. A blueprint untouched while the repo keeps moving is logged as a `DRIFT` assumption.

2. **Structural drift (plan vs. files)** — reads each blueprint's nodes and, for any node that declares a file it represents (a `path` string or `paths` list inside the node's free-form `data` dict — see `Node.data` in T1.3.03), checks whether that file actually exists on disk under the project root. A declared-but-missing file is logged as a `MISSING` assumption. This is what turns the Observer from a timestamp checker into a structural integrity validator: it answers "does the plan still match the code?", not just "is the plan old?".

The script is a standalone CLI tool run by Hermes or manually. It uses `httpx` from the backend venv, reads `CELESTIAL_API_URL` and `CELESTIAL_AUTH_TOKEN` from the environment, and resolves node file paths relative to a project root (`--root`, or `CELESTIAL_PROJECT_ROOT`, or the git toplevel, or the cwd).

## Objective

Create `<project_root>/tools/observer.py` — a CLI script that detects stale active blueprints (timestamp drift) **and** blueprint nodes pointing at files that no longer exist (structural drift), logging a drift assumption for each finding via the Celestial API.

   ```python
   #!/usr/bin/env python3
   """
   Celestial Blueprint Observer

   Detects two kinds of drift in active blueprints and logs an assumption for each:
     - DRIFT  : blueprint not updated while the project keeps getting git commits
     - MISSING: a node declares a file (data.path / data.paths) that no longer exists

   Usage:
     python3 tools/observer.py                      # check all active blueprints
     python3 tools/observer.py --id <bp-id>         # check one blueprint
     python3 tools/observer.py --dry-run            # print findings, no writes
     python3 tools/observer.py --stale-days 7       # staleness threshold (default: 3)
     python3 tools/observer.py --root /path/to/repo # root for resolving node paths
   """
   from __future__ import annotations

   import argparse
   import os
   import subprocess
   from datetime import datetime, timezone
   from pathlib import Path

   import httpx

   API_URL: str = os.environ.get("CELESTIAL_API_URL", "http://localhost:8787")
   TOKEN: str = os.environ.get("CELESTIAL_AUTH_TOKEN", "")
   HEADERS: dict[str, str] = {"Authorization": f"Bearer {TOKEN}"} if TOKEN else {}


   def git_toplevel() -> Path | None:
       """Return the git repo root, or None if not in a repo."""
       try:
           out = subprocess.check_output(
               ["git", "rev-parse", "--show-toplevel"],
               stderr=subprocess.DEVNULL,
               text=True,
           ).strip()
           return Path(out) if out else None
       except Exception:
           return None


   def latest_git_commit_ts() -> datetime | None:
       """Return UTC datetime of the most recent git commit in the repo root."""
       try:
           out = subprocess.check_output(
               ["git", "log", "-1", "--format=%ct"],
               stderr=subprocess.DEVNULL,
               text=True,
           ).strip()
           return datetime.fromtimestamp(int(out), tz=timezone.utc) if out else None
       except Exception:
           return None


   def staleness_days(updated_at: str, now: datetime) -> float:
       """Return how many days ago updated_at was. updated_at is ISO-8601 or date."""
       try:
           dt = datetime.fromisoformat(updated_at.replace("Z", "+00:00"))
           if dt.tzinfo is None:
               dt = dt.replace(tzinfo=timezone.utc)
           return (now - dt).total_seconds() / 86400
       except Exception:
           return 0.0


   def declared_paths(node: dict) -> list[str]:
       """Extract file paths a node claims to represent, from its free-form data dict."""
       data = node.get("data") or {}
       paths: list[str] = []
       one = data.get("path")
       if isinstance(one, str) and one.strip():
           paths.append(one.strip())
       many = data.get("paths")
       if isinstance(many, list):
           paths.extend(p.strip() for p in many if isinstance(p, str) and p.strip())
       return paths


   def missing_files(bp: dict, root: Path) -> list[tuple[str, str]]:
       """Return (node_id, path) pairs for every declared file that does not exist."""
       out: list[tuple[str, str]] = []
       for node in bp.get("nodes", []):
           for rel in declared_paths(node):
               target = Path(rel) if os.path.isabs(rel) else (root / rel)
               if not target.exists():
                   out.append((node.get("id", "?"), rel))
       return out


   def log_assumption(bid: str, aid: str, body: str) -> None:
       """Record a drift assumption against a blueprint via the API."""
       httpx.post(
           f"{API_URL}/api/blueprints/{bid}/assumptions",
           json={"id": aid, "status": "unvalidated", "risk": "high", "body": body},
           headers=HEADERS,
           timeout=15,
       )


   def observe_blueprint(
       bid: str, dry_run: bool, stale_days: float, now: datetime,
       commit_ts: datetime | None, root: Path,
   ) -> list[str]:
       """Check one blueprint for staleness + structural drift; return finding strings."""
       r = httpx.get(f"{API_URL}/api/blueprints/{bid}", headers=HEADERS, timeout=15)
       if r.status_code != 200:
           return [f"  SKIP {bid}: {r.status_code}"]
       bp = r.json()
       fm = bp.get("frontmatter", {})
       if fm.get("status", "") not in ("draft", "active", "in-progress"):
           return []

       findings: list[str] = []
       stamp = now.strftime("%Y%m%d")

       # 1) staleness / timestamp drift
       updated = fm.get("modified", fm.get("date_created", ""))
       days = staleness_days(updated, now) if updated else 0.0
       if days >= stale_days and commit_ts is not None:
           msg = f"Reality drift — blueprint not updated in {days:.0f} days but project has recent commits"
           findings.append(f"  DRIFT {bid}: {msg}")
           if not dry_run:
               log_assumption(bid, f"obs-drift-{stamp}", msg)

       # 2) structural drift — nodes pointing at files that no longer exist
       for nid, rel in missing_files(bp, root):
           msg = f"Reality drift — node '{nid}' references missing file: {rel}"
           findings.append(f"  MISSING {bid}: {msg}")
           if not dry_run:
               log_assumption(bid, f"obs-missing-{nid}-{stamp}", msg)

       return findings


   def main() -> None:
       parser = argparse.ArgumentParser(description="Celestial Blueprint Observer")
       parser.add_argument("--id", dest="bp_id", default="", help="Check one blueprint ID")
       parser.add_argument("--dry-run", action="store_true", help="Print findings, no writes")
       parser.add_argument("--stale-days", type=float, default=3.0, help="Staleness threshold in days (default 3)")
       parser.add_argument("--root", default=os.environ.get("CELESTIAL_PROJECT_ROOT", ""),
                           help="Root for resolving node file paths (default: git toplevel or cwd)")
       args = parser.parse_args()

       now = datetime.now(tz=timezone.utc)
       commit_ts = latest_git_commit_ts()
       root = Path(args.root) if args.root else (git_toplevel() or Path.cwd())

       if args.bp_id:
           bps = [{"id": args.bp_id}]
       else:
           resp = httpx.get(f"{API_URL}/api/blueprints", headers=HEADERS, timeout=15)
           resp.raise_for_status()
           bps = resp.json()

       total = 0
       for bp in bps:
           bid = bp.get("id", bp.get("frontmatter", {}).get("id", ""))
           for f in observe_blueprint(bid, args.dry_run, args.stale_days, now, commit_ts, root):
               print(f)
               total += 1

       mode = "DRY-RUN " if args.dry_run else ""
       print(f"Observer {mode}complete: {len(bps)} blueprint(s) checked, {total} finding(s), root={root}")


   if __name__ == "__main__":
       main()
   ```
