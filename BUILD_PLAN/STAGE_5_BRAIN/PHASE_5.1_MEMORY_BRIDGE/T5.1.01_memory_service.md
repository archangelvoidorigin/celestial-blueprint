The brain-orchestrator lives at `03_TOOLBOX/tools/brain-regions/orchestrator.py` and speaks the same raw stdio JSON-RPC protocol as the Celestial MCP server (T4.2.01). `memory.py` calls it via subprocess to fetch context snippets for a given topic, seeding research nodes during planning. ## Objective

3. Create `backend/app/services/memory.py` with exactly:
   ```python
   """Memory bridge: query the brain-orchestrator for context relevant to a topic."""
   from __future__ import annotations

   import json
   import os
   import subprocess
   from typing import Any

   from app.core.config import settings


   def query_memory(topic: str, limit: int = 5) -> list[dict[str, Any]]:
       """Return up to `limit` memory snippets relevant to `topic`.

       Calls the brain-orchestrator at settings.brain_orchestrator_path via
       JSON-RPC stdio (tools/call → brain_retrieve). Returns [] if not
       configured, path does not exist, or the call fails for any reason.
       """
       path = settings.brain_orchestrator_path
       if not path or not os.path.exists(path):
           return []

       request = json.dumps({
           "jsonrpc": "2.0",
           "id": 1,
           "method": "tools/call",
           "params": {
               "name": "brain_retrieve",
               "arguments": {"query": topic, "limit": limit},
           },
       })

       try:
           result = subprocess.run(
               ["python3", path],
               input=request,
               capture_output=True,
               text=True,
               timeout=15,
           )
           first_line = result.stdout.strip().splitlines()[0] if result.stdout.strip() else ""
           if not first_line:
               return []
           resp = json.loads(first_line)
           content = resp.get("result", {}).get("content", [])
           return [{"text": c.get("text", ""), "type": c.get("type", "text")} for c in content if c.get("text")]
       except Exception:
           return []
   ```
