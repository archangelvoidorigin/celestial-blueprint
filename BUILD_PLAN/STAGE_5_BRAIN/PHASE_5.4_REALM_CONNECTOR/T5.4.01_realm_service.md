## Context

Blueprints have a `realm` frontmatter field (e.g. `"00_CORTEX/01_CREATION_REALM/Celestial_Blueprint"`). This is a path relative to `CELESTIAL_FORGE_ROOT` (`/home/void-yggdrasil/CHAOS_DAO_FORGE`). `realm.py` resolves this path, reads the realm's `smart-folder.md`, parses its YAML frontmatter, and returns a context dict (phase, agents, description, last-modified date). If the path does not resolve or smart-folder.md is absent, the service returns `None` — never crashing the blueprint GET.

   ```python
   """Realm connector: read a CHAOS_DAO_FORGE realm's smart-folder.md for context."""
   from __future__ import annotations

   from pathlib import Path
   from typing import Any

   import yaml

   from app.core.config import settings


   def get_realm_context(realm: str) -> dict[str, Any] | None:
       """Return context dict from the realm's smart-folder.md, or None if unavailable.

       `realm` is either:
         - An absolute path (starts with '/')
         - A path relative to settings.forge_root
       """
       if not realm:
           return None

       if realm.startswith("/"):
           base = Path(realm)
       else:
           base = Path(settings.forge_root) / realm

       smart_folder = base / "smart-folder.md"
       if not smart_folder.exists():
           return None

       try:
           text = smart_folder.read_text(encoding="utf-8")
           # extract YAML frontmatter between --- fences
           if text.startswith("---"):
               parts = text.split("---", 2)
               fm = yaml.safe_load(parts[1]) if len(parts) >= 3 else {}
           else:
               fm = {}
           stat = smart_folder.stat()
           from datetime import datetime
           return {
               "realm_path": str(base),
               "title": fm.get("title", ""),
               "phase": fm.get("phase", fm.get("current_phase", "")),
               "status": fm.get("status", ""),
               "agents": fm.get("agents", []),
               "tags": fm.get("tags", []),
               "last_modified": datetime.fromtimestamp(stat.st_mtime).isoformat(timespec="seconds"),
           }
       except Exception:
           return None
   ```
