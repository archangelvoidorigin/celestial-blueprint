## Context

The MCP server is a Python script at `mcp/server.py` that speaks the MCP stdio JSON-RPC protocol — the same raw protocol used by `03_TOOLBOX/tools/brain-regions/orchestrator.py`. It does NOT use the `mcp` Python library (not installed). It handles three methods: `initialize`, `tools/list`, and `tools/call`. All tool implementations are thin HTTP calls to the running Celestial backend at `CELESTIAL_API_URL` (default `http://localhost:8787`). The backend venv (`backend/.venv/bin/python3`) is used as the runtime — it already has `httpx` installed.

Exposed tools (14 total):

| Tool | Maps to |
|------|---------|
| `blueprint_list` | GET /api/blueprints |
| `blueprint_get` | GET /api/blueprints/{id} |
| `blueprint_create` | POST /api/blueprints |
| `blueprint_patch` | PATCH /api/blueprints/{id} |
| `blueprint_archive` | POST /api/blueprints/{id}/archive |

   """
   Celestial Blueprint MCP Server

   Exposes the Celestial Blueprint REST API as an MCP tool surface over stdio.
   All API calls go through the running backend at CELESTIAL_API_URL.

   Transport: raw stdio JSON-RPC (same pattern as brain-regions/orchestrator.py).

     python3 mcp/server.py         # Start MCP server (stdio transport)

   API_URL: str = os.environ.get("CELESTIAL_API_URL", "http://localhost:8787")

       r = httpx.post(f"{API_URL}{path}", json=body or {}, headers=HEADERS, timeout=30)
       r.raise_for_status()
       return r.json()

       r = httpx.patch(f"{API_URL}{path}", json=body, headers=HEADERS, timeout=30)

   # ── Tool dispatch (tools/call) ─────────────────────────────────────────

   def call_tool(name: str, args: dict) -> Any:
       if name == "blueprint_list":
           return _get("/api/blueprints")

       elif name == "blueprint_get":
           return _get(f"/api/blueprints/{args['id']}")

       elif name == "blueprint_create":
           body: dict = {"name": args["name"]}
           for k in ("tier", "realm", "summary"):
               if k in args:
                   body[k] = args[k]
           return _post("/api/blueprints", body)

       elif name == "blueprint_patch":
           bp_id = args["id"]
           body = {k: v for k, v in args.items() if k != "id" and v is not None}
           return _patch(f"/api/blueprints/{bp_id}", body)

       elif name == "blueprint_archive":
           return _post(f"/api/blueprints/{args['id']}/archive")

       elif name == "node_add":
           bp_id = args["blueprint_id"]

           for k in ("cat", "id", "body"):

   # ── stdio MCP loop ──────────────────────────────────────────────────────

   def run_server() -> None:
       """Read JSON-RPC requests from stdin, write responses to stdout."""
       while True:
           line = sys.stdin.readline()
           if not line:
               break
           line = line.strip()
           if not line:
               continue

           try:
               req = json.loads(line)
           except json.JSONDecodeError:
               print(json.dumps({
                   "jsonrpc": "2.0", "id": None,
                   "error": {"code": -32700, "message": "Parse error"},
               }))
               sys.stdout.flush()
               continue

           req_id = req.get("id")
           method = req.get("method", "")
           params = req.get("params", {})
           response: dict = {"jsonrpc": "2.0", "id": req_id}

           try:
               if method == "initialize":
                   response["result"] = {
                       "protocolVersion": "2024-11-05",
                       "capabilities": {"tools": {}},
                       "serverInfo": {"name": "celestial-blueprint", "version": "1.0.0"},
                   }

               elif method == "tools/list":
                   response["result"] = {"tools": TOOLS}

               elif method == "tools/call":
                   tool_name = params.get("name", "")
                   args = params.get("arguments", {})
                   result = call_tool(tool_name, args)
                   text = result if isinstance(result, str) else json.dumps(result, indent=2)
                   response["result"] = {"content": [{"type": "text", "text": text}]}

               elif method.startswith("notifications/"):
                   # Notifications have no response
                   continue

               else:
                   response["error"] = {
                       "code": -32601,
                       "message": f"Method not found: {method}",
                   }

           except httpx.HTTPStatusError as exc:
               response["error"] = {
                   "code": -32603,
                   "message": f"API error {exc.response.status_code}: {exc.response.text}",
               }
           except Exception as exc:
               response["error"] = {"code": -32603, "message": str(exc)}

           print(json.dumps(response))
           sys.stdout.flush()


   if __name__ == "__main__":
       if "--test" in sys.argv:
           print(f"Tools available: {len(TOOLS)}")
           for t in TOOLS:
               print(f"  {t['name']}: {t['description'][:60]}")
           sys.exit(0)
       run_server()
