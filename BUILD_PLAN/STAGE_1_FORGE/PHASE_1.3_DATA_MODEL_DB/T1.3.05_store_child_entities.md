   ```python
   def _touch(c, bid: str) -> None:
       c.execute("UPDATE blueprints SET updated_at = ? WHERE id = ?", (_now(), bid))


   # ---- incremental adders (INSERT OR REPLACE so they upsert by id) ----

   def add_node(bid: str, node: dict) -> None:
       with connection() as c:
           c.execute(
               """INSERT OR REPLACE INTO nodes
                  (blueprint_id, id, cat, type, title, body, data_json)
                  VALUES (?,?,?,?,?,?,?)""",
               (bid, node["id"], node.get("cat", ""), node["type"],

           )
           _touch(c, bid)


   def add_edge(bid: str, edge: dict) -> None:
       with connection() as c:
           c.execute(
               "INSERT INTO edges (blueprint_id, src, dst, label) VALUES (?,?,?,?)",
               (bid, edge["src"], edge["dst"], edge.get("label", "")),

   def list_chronicles(bid: str) -> list[dict]:
       with connection() as c:
           rows = c.execute(
               "SELECT * FROM chronicles WHERE blueprint_id = ? ORDER BY id", (bid,)
           ).fetchall()
       return [dict(r) for r in rows]

   def _replace_children(c, bid: str, bp: dict) -> None:
       for table in ("nodes", "edges", "assumptions", "decisions", "risks", "versions"):
           c.execute(f"DELETE FROM {table} WHERE blueprint_id = ?", (bid,))
       for n in bp.get("nodes", []):
           c.execute(
               """INSERT INTO nodes (blueprint_id, id, cat, type, title, body, data_json)
                  VALUES (?,?,?,?,?,?,?)""",
               (bid, n["id"], n.get("cat", ""), n["type"], n.get("title", ""),
                n.get("body", ""), json.dumps(n.get("data", {}))),
           )
       for e in bp.get("edges", []):
           c.execute("INSERT INTO edges (blueprint_id, src, dst, label) VALUES (?,?,?,?)",
                     (bid, e["src"], e["dst"], e.get("label", "")))
       for a in bp.get("assumptions", []):
           c.execute(
               """INSERT INTO assumptions (blueprint_id, id, bp_ref, status, risk, body)
                  VALUES (?,?,?,?,?,?)""",
               (bid, a["id"], a.get("bp_ref", ""), a.get("status", ""),
                a.get("risk", ""), a.get("body", "")))
       for d in bp.get("decisions", []):
           c.execute(
               """INSERT INTO decisions (blueprint_id, id, bp_ref, status, date, body)
                  VALUES (?,?,?,?,?,?)""",
               (bid, d["id"], d.get("bp_ref", ""), d.get("status", ""),
                d.get("date", ""), d.get("body", "")))
       for r in bp.get("risks", []):
           c.execute(
               """INSERT INTO risks (blueprint_id, id, risk, likelihood, impact, mitigation, status)
                  VALUES (?,?,?,?,?,?,?)""",
               (bid, r["id"], r.get("risk", ""), r.get("likelihood", ""),
                r.get("impact", ""), r.get("mitigation", ""), r.get("status", "")))
       for v in bp.get("versions", []):
           c.execute("INSERT INTO versions (blueprint_id, version, date, note) VALUES (?,?,?,?)",
                     (bid, v["version"], v.get("date", ""), v.get("note", "")))


   def save_blueprint(bp: dict) -> str:
       """Full upsert of a canonical blueprint dict. Used by markdown import."""
       fm = bp["frontmatter"]
       bid = fm["id"]
       with connection() as c:
           c.execute(
               """INSERT OR REPLACE INTO blueprints
                  (id, name, version, tier, status, realm, summary,
                   frontmatter_json, categories_json, research_json,
                   chronicle_links_json, created_at, updated_at)
                  VALUES (?,?,?,?,?,?,?,?,?,?,?,
                          COALESCE((SELECT created_at FROM blueprints WHERE id = ?), ?),
                          ?)""",
               (bid, fm.get("name", ""), fm.get("version", "0.1.0"),
                fm.get("tier", "small"), fm.get("status", "draft"),
                fm.get("realm", ""), bp.get("summary", ""),
                json.dumps(fm), json.dumps(bp.get("categories", [])),
                json.dumps(bp.get("research", [])),
                json.dumps(bp.get("chronicle_links", [])),
                bid, _now(), _now()),
           )
           _replace_children(c, bid, bp)
       return bid


   def get_blueprint(bid: str) -> dict | None:
       """Assemble the full canonical blueprint dict from all tables."""
       meta = get_blueprint_meta(bid)
       if meta is None:
           return None
       with connection() as c:
           nodes = [dict(r) for r in c.execute(
               "SELECT id, cat, type, title, body, data_json FROM nodes WHERE blueprint_id = ?", (bid,))]
           edges = [dict(r) for r in c.execute(
               "SELECT src, dst, label FROM edges WHERE blueprint_id = ?", (bid,))]
           assumptions = [dict(r) for r in c.execute(
               "SELECT id, bp_ref, status, risk, body FROM assumptions WHERE blueprint_id = ?", (bid,))]
           decisions = [dict(r) for r in c.execute(
               "SELECT id, bp_ref, status, date, body FROM decisions WHERE blueprint_id = ?", (bid,))]
           risks = [dict(r) for r in c.execute(
               "SELECT id, risk, likelihood, impact, mitigation, status FROM risks WHERE blueprint_id = ?", (bid,))]
           versions = [dict(r) for r in c.execute(
               "SELECT version, date, note FROM versions WHERE blueprint_id = ?", (bid,))]
       for n in nodes:
           n["data"] = json.loads(n.pop("data_json"))
       return {
           "frontmatter": meta["frontmatter"],
           "summary": meta["summary"],
           "categories": meta["categories"],
           "nodes": nodes,
           "assumptions": assumptions,
           "decisions": decisions,
           "risks": risks,
           "research": meta["research"],
           "chronicle_links": meta["chronicle_links"],
           "edges": edges,
           "versions": versions,
       }


   def archive_blueprint(bid: str) -> bool:
       """Never delete: flip status to 'archived'."""
       with connection() as c:
           cur = c.execute(
               "UPDATE blueprints SET status = 'archived', updated_at = ? WHERE id = ?",
               (_now(), bid))
           return cur.rowcount > 0
   ```
