   ```python
   """Blueprint persistence (DB index). The only module that touches SQL."""
   from __future__ import annotations

   import json
   import re
   from datetime import date, datetime

   from app.core.db import connection
   from app.models.schemas import BlueprintCreate, BlueprintPatch

   TIER_MODEL = {"small": "haiku", "medium": "sonnet",
                 "large": "opus", "continuous": "adaptive"}
   TIER_ITER = {"small": 3, "medium": 2, "large": 1, "continuous": 0}


   def _now() -> str:
       return datetime.now().isoformat(timespec="seconds")


   def slugify(name: str) -> str:
       s = re.sub(r"[^a-z0-9]+", "-", name.lower()).strip("-")
       return s or "blueprint"


   def make_blueprint_id(name: str, day: str | None = None) -> str:
       day = day or date.today().isoformat()
       return f"bp_{day}_{slugify(name)}"


   def default_frontmatter(bid: str, data: BlueprintCreate, day: str) -> dict:
       return {
           "id": bid,
           "name": data.name,
           "version": "0.1.0",
           "tier": data.tier,
           "status": "draft",
           "realm": data.realm,
           "title": data.name,
           "tags": ["planning", "blueprint"],
           "pipeline_stage": "ideation",
           "date_created": day,
           "modified": day,
           "model": TIER_MODEL.get(data.tier, "haiku"),
           "iterations": TIER_ITER.get(data.tier, 3),
       }


   def blueprint_exists(bid: str) -> bool:
       with connection() as c:
           return c.execute(
               "SELECT 1 FROM blueprints WHERE id = ?", (bid,)
           ).fetchone() is not None


   def create_blueprint(data: BlueprintCreate) -> str:
       day = date.today().isoformat()
       bid = make_blueprint_id(data.name, day)
       fm = default_frontmatter(bid, data, day)
       with connection() as c:
           c.execute(
               """INSERT INTO blueprints
                  (id, name, version, tier, status, realm, summary,
                   frontmatter_json, created_at, updated_at)
                  VALUES (?,?,?,?,?,?,?,?,?,?)""",
               (bid, data.name, "0.1.0", data.tier, "draft", data.realm,
                data.summary, json.dumps(fm), _now(), _now()),
           )
       return bid


   def get_blueprint_meta(bid: str) -> dict | None:
       """Blueprint-level fields only (no child entities). None if absent."""
       with connection() as c:
           row = c.execute(
               "SELECT * FROM blueprints WHERE id = ?", (bid,)
           ).fetchone()
       if row is None:
           return None
       return {
           "frontmatter": json.loads(row["frontmatter_json"]),
           "summary": row["summary"] or "",
           "categories": json.loads(row["categories_json"]),
           "research": json.loads(row["research_json"]),
           "chronicle_links": json.loads(row["chronicle_links_json"]),
           "updated_at": row["updated_at"],
       }


   def list_blueprints() -> list[dict]:
       with connection() as c:
           rows = c.execute(
               """SELECT id, name, version, tier, status, updated_at
                  FROM blueprints ORDER BY updated_at DESC"""
           ).fetchall()
       return [dict(r) for r in rows]


   def patch_blueprint(bid: str, patch: BlueprintPatch) -> bool:
       meta = get_blueprint_meta(bid)
       if meta is None:
           return False
       fm = meta["frontmatter"]
       fields = patch.model_dump(exclude_none=True)
       for k, v in fields.items():
           fm[k] = v
       fm["modified"] = date.today().isoformat()
       with connection() as c:
           c.execute(
               """UPDATE blueprints SET
                    name = COALESCE(?, name),
                    status = COALESCE(?, status),
                    version = COALESCE(?, version),
                    summary = COALESCE(?, summary),
                    frontmatter_json = ?, updated_at = ?
                  WHERE id = ?""",
               (fields.get("name"), fields.get("status"), fields.get("version"),
                fields.get("summary"), json.dumps(fm), _now(), bid),
           )
       return True
   ```
