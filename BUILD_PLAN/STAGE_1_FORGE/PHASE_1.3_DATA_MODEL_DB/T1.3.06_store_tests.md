## Instructions
1. Create `backend/tests/test_store.py` with exactly:
   ```python
   """Store layer tests against an isolated temp DB."""
   import importlib
   from pathlib import Path

   import pytest

   TEMPLATES = Path(__file__).resolve().parents[2] / "docs" / "templates"


   @pytest.fixture()
   def store(tmp_path, monkeypatch):
       """Fresh DB + freshly imported store/config bound to it."""
       db = tmp_path / "t.db"
       monkeypatch.setenv("CELESTIAL_DB_PATH", str(db))
       import app.core.config as config
       importlib.reload(config)
       import app.core.db as db_mod
       importlib.reload(db_mod)
       import app.services.store as store_mod
       importlib.reload(store_mod)
       db_mod.init_db()
       return store_mod


   def test_create_and_get_meta(store):
       from app.models.schemas import BlueprintCreate
       bid = store.create_blueprint(BlueprintCreate(name="My Plan", tier="small"))
       assert bid.endswith("my-plan")
       meta = store.get_blueprint_meta(bid)
       assert meta["frontmatter"]["model"] == "haiku"


   def test_patch(store):
       from app.models.schemas import BlueprintCreate, BlueprintPatch
       bid = store.create_blueprint(BlueprintCreate(name="P"))
       assert store.patch_blueprint(bid, BlueprintPatch(status="active"))
       assert store.get_blueprint_meta(bid)["frontmatter"]["status"] == "active"


   def test_full_roundtrip(store):
       from app.services.parser import parse_blueprint
       bp = parse_blueprint((TEMPLATES / "small.md").read_text())
       bid = store.save_blueprint(bp)
       got = store.get_blueprint(bid)
       assert len(got["nodes"]) == len(bp["nodes"])
       assert len(got["edges"]) == len(bp["edges"])
       assert len(got["assumptions"]) == len(bp["assumptions"])


   def test_add_node(store):
       from app.models.schemas import BlueprintCreate
       bid = store.create_blueprint(BlueprintCreate(name="N"))
       store.add_node(bid, {"id": "n1", "cat": "c1", "type": "text",
                            "title": "T", "body": "b", "data": {}})
       assert len(store.get_blueprint(bid)["nodes"]) == 1


   def test_archive_never_deletes(store):
       from app.models.schemas import BlueprintCreate
       bid = store.create_blueprint(BlueprintCreate(name="A"))
       assert store.archive_blueprint(bid)
       meta = store.get_blueprint_meta(bid)
       assert meta is not None
       assert meta["frontmatter"]["status"] == "draft"  # frontmatter copy unchanged
       assert store.list_blueprints()[0]["status"] == "archived"  # column flipped
   ```
