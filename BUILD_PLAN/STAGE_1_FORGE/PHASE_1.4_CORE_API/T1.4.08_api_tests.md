   ```python
   """API integration tests through TestClient on an isolated DB."""
   import glob
   import importlib
   import os

   import pytest


   @pytest.fixture()
   def client(tmp_path, monkeypatch):
       monkeypatch.setenv("CELESTIAL_DB_PATH", str(tmp_path / "api.db"))
       monkeypatch.setenv("CELESTIAL_BLUEPRINTS_DIR", str(tmp_path / "bps"))
       monkeypatch.setenv("CELESTIAL_CHRONICLES_DIR", str(tmp_path / "chron"))
       import app.core.config as config
       importlib.reload(config)
       import app.core.db as db_mod
       importlib.reload(db_mod)
       import app.services.store as store_mod
       importlib.reload(store_mod)
       import app.services.files as files_mod
       importlib.reload(files_mod)
       from fastapi.testclient import TestClient
       import app.main as main_mod
       importlib.reload(main_mod)
       with TestClient(main_mod.app) as c:
           yield c


   def _new(client, name="Plan", tier="small"):
       r = client.post("/api/blueprints", json={"name": name, "tier": tier})
       assert r.status_code == 200, r.text
       return r.json()["frontmatter"]["id"]


   def test_health(client):
       assert client.get("/api/health").json()["status"] == "ok"


   def test_blueprint_lifecycle(client):
       bid = _new(client)
       assert client.get(f"/api/blueprints/{bid}").status_code == 200
       assert client.get("/api/blueprints").json()
       r = client.patch(f"/api/blueprints/{bid}", json={"status": "active"})
       assert r.json()["frontmatter"]["status"] == "active"
       assert client.post(f"/api/blueprints/{bid}/archive").json()["status"] == "archived"


   def test_unknown_404(client):
       assert client.get("/api/blueprints/nope").status_code == 404


   def test_nodes_and_tracking(client):
       bid = _new(client, tier="medium")
       client.post(f"/api/blueprints/{bid}/nodes",
                   json={"id": "n1", "type": "text", "title": "T"})
       client.post(f"/api/blueprints/{bid}/assumptions",
                   json={"id": "a1", "risk": "high"})
       bp = client.get(f"/api/blueprints/{bid}").json()
       assert len(bp["nodes"]) == 1 and len(bp["assumptions"]) == 1


   def test_chronicle(client):
       bid = _new(client)
       r = client.post(f"/api/blueprints/{bid}/chronicles",
                       json={"kind": "raw", "body": "note"})
       assert os.path.exists(r.json()["path"])
       assert len(client.get(f"/api/blueprints/{bid}/chronicles").json()) == 1


   def test_export_and_rebuild_recovery(client, tmp_path):
       bid = _new(client, name="Recover")
       md = client.get(f"/api/blueprints/{bid}/export").text
       assert md.startswith("---")
       for f in glob.glob(str(tmp_path / "api.db*")):
           os.remove(f)
       assert client.post("/api/rebuild").json()["imported"] >= 1
       assert client.get(f"/api/blueprints/{bid}").status_code == 200
   ```
