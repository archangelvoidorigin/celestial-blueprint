   """Celestial Blueprint API. All planning logic lives behind this app."""
   from __future__ import annotations

   from contextlib import asynccontextmanager

   from fastapi import FastAPI
   from fastapi.middleware.cors import CORSMiddleware

   from app.core.config import settings
   from app.core.db import init_db


   @asynccontextmanager
   async def lifespan(app: FastAPI):
       init_db()
       yield


   app = FastAPI(title="Celestial Blueprint", version="0.1.0", lifespan=lifespan)

   app.add_middleware(
       CORSMiddleware,
       allow_origins=settings.cors_origins,
       allow_credentials=True,
       allow_methods=["*"],
       allow_headers=["*"],
   )


   @app.get("/api/health")
   def health() -> dict:
       return {"status": "ok", "service": "celestial-blueprint"}


   # --- ROUTER INCLUDES (added by T1.4.02–T1.4.06) ---
