## Context
The backend is a Python package rooted at `backend/app`. Python needs `__init__.py` files so `from app.core.config import settings` works. Configuration (DB path, dirs, auth token, port, CORS origins) is read once from environment variables (loaded from `.env`) into a single `settings` object that every other module imports. The env var names were fixed in T1.1.03 (`CELESTIAL_DB_PATH`, `CELESTIAL_BLUEPRINTS_DIR`, `CELESTIAL_CHRONICLES_DIR`, `CELESTIAL_AUTH_TOKEN`, `CELESTIAL_PORT`, `CELESTIAL_CORS_ORIGINS`).

## Objective

2. Create `backend/app/core/config.py` with exactly:
   ```python
   """Central configuration. Reads from environment (loaded from .env)."""
   from __future__ import annotations

   import os
   from dataclasses import dataclass, field
   from pathlib import Path


   def _load_dotenv(path: Path) -> None:
       """Minimal .env loader (no external dependency). KEY=VALUE per line."""
       if not path.exists():
           return
       for raw in path.read_text().splitlines():
           line = raw.strip()
           if not line or line.startswith("#") or "=" not in line:
               continue
           key, _, value = line.partition("=")
           key, value = key.strip(), value.strip().strip('"').strip("'")
           os.environ.setdefault(key, value)


   # Project root = backend/ (two parents up from this file: core -> app -> backend)
   BACKEND_DIR = Path(__file__).resolve().parents[2]
   PROJECT_ROOT = BACKEND_DIR.parent
   _load_dotenv(BACKEND_DIR / ".env")


   def _csv(value: str) -> list[str]:
       return [v.strip() for v in value.split(",") if v.strip()]


   @dataclass(frozen=True)
   class Settings:
       db_path: str = field(
           default_factory=lambda: os.environ.get(
               "CELESTIAL_DB_PATH", str(PROJECT_ROOT / "celestial.db")
           )
       )
       blueprints_dir: str = field(
           default_factory=lambda: os.environ.get(
               "CELESTIAL_BLUEPRINTS_DIR", str(PROJECT_ROOT / "blueprints")
           )
       )
       chronicles_dir: str = field(
           default_factory=lambda: os.environ.get(
               "CELESTIAL_CHRONICLES_DIR", str(PROJECT_ROOT / "chronicles")
           )
       )
       auth_token: str = field(
           default_factory=lambda: os.environ.get("CELESTIAL_AUTH_TOKEN", "")
       )
       port: int = field(
           default_factory=lambda: int(os.environ.get("CELESTIAL_PORT", "8787"))
       )
       cors_origins: list[str] = field(
           default_factory=lambda: _csv(
               os.environ.get("CELESTIAL_CORS_ORIGINS", "http://localhost:5173")
           )
       )


   settings = Settings()
   ```
