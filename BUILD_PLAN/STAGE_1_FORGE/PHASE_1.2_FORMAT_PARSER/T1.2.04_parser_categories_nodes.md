## Context
T1.2.03 created `parser.py` with frontmatter + anchor helpers and the canonical blueprint dict. This task adds the readers for the three core body sections: the `<!-- bp:summary -->` paragraph, the `## Categories` list, and the `## Nodes` blocks. Each node is a `### Title` heading, then a `<!-- node:TYPE id=.. cat=.. ...extra -->` anchor, then prose until the next `###`/`##`. Base anchor fields (`id`, `cat`, `type`) become top-level; all extra fields go into the node's `data` dict (forward-compatible per `docs/FORMAT.md`).

## Objective

   ```python
   _BASE_NODE_FIELDS = {"id", "cat", "type"}


   def _section(body: str, name: str) -> str:
       """Return the text under '## {name}' up to the next '## ' (exclusive)."""
       pat = re.compile(rf"^##\s+{re.escape(name)}\s*$", re.MULTILINE)
       m = pat.search(body)
       if not m:
           return ""
       start = m.end()
       nxt = re.compile(r"^##\s+", re.MULTILINE).search(body, start)
       return body[start: nxt.start() if nxt else len(body)]


   def parse_summary(body: str) -> str:
       """Prose paragraph following the <!-- bp:summary --> anchor."""
       m = re.search(r"<!--\s*bp:summary\s*-->", body)
       if not m:
           return ""
       after = body[m.end():]
       # up to the first '## ' heading
       nxt = re.compile(r"^##\s+", re.MULTILINE).search(after)
       chunk = after[: nxt.start() if nxt else len(after)]
       return chunk.strip()


   def parse_categories(body: str) -> list[dict]:
       """Parse '- <!-- cat:c1 --> Name' lines in the Categories section."""
       sec = _section(body, "Categories")
       out = []
       for cid, name in re.findall(
           r"<!--\s*cat:(\w+)\s*-->\s*(.*)", sec
       ):
           out.append({"id": cid, "name": name.strip()})
       return out


   def parse_nodes(body: str) -> list[dict]:
       """Parse '### Title\\n<!-- node:TYPE id=.. cat=.. ... -->\\nprose' blocks."""
       sec = _section(body, "Nodes")
       nodes = []
       # Split into '### ' blocks
       blocks = re.split(r"^###\s+", sec, flags=re.MULTILINE)
       for blk in blocks:
           blk = blk.strip()
           if not blk:
               continue
           lines = blk.split("\n", 1)
           title = lines[0].strip()
           remainder = lines[1] if len(lines) > 1 else ""
           m = re.search(r"<!--\s*node:(\w+)\s+(.*?)-->", remainder, re.DOTALL)
           if not m:
               continue
           ntype = m.group(1)
           attrs = parse_anchor_attrs(m.group(2))
           prose = remainder[m.end():].strip()
           data = {k: v for k, v in attrs.items() if k not in _BASE_NODE_FIELDS}
           nodes.append({
               "id": attrs.get("id", ""),
               "cat": attrs.get("cat", ""),
               "type": ntype,
               "title": title,
               "body": prose,
               "data": data,
           })
       return nodes
   ```
