   ```python
   def _list_anchor_items(sec: str, kind: str):
       """Yield (id, attrs_dict, prose) for '- <!-- kind:id attrs --> prose' items."""
       pat = re.compile(
           rf"-\s*<!--\s*{kind}:(\w+)\s+(.*?)-->(.*?)(?=\n-\s*<!--|\Z)",
           re.DOTALL,
       )
       for m in pat.finditer(sec):
           yield m.group(1), parse_anchor_attrs(m.group(2)), m.group(3).strip()


   def parse_assumptions(body: str) -> list[dict]:
       sec = _section(body, "Assumptions")
       return [
           {"id": i, "bp_ref": a.get("bp_ref", ""), "status": a.get("status", ""),
            "risk": a.get("risk", ""), "body": p}
           for i, a, p in _list_anchor_items(sec, "assumption")
       ]


   def parse_decisions(body: str) -> list[dict]:
       sec = _section(body, "Decisions")
       return [
           {"id": i, "bp_ref": a.get("bp_ref", ""), "status": a.get("status", ""),
            "date": a.get("date", ""), "body": p}
           for i, a, p in _list_anchor_items(sec, "decision")
       ]


   def parse_research(body: str) -> list[dict]:
       sec = _section(body, "Research Log")
       return [
           {"id": i, "query": a.get("query", ""), "date": a.get("date", ""), "body": p}
           for i, a, p in _list_anchor_items(sec, "research")
       ]


   def parse_chronicle_links(body: str) -> list[dict]:
       sec = _section(body, "Chronicle Links")
       out = []
       for m in re.finditer(
           r"-\s*<!--\s*chronicle:(\w+)\s+type=(\w+)\s*-->\s*(\S+)", sec
       ):
           out.append({"id": m.group(1), "type": m.group(2), "path": m.group(3)})
       return out


   def parse_risks(body: str) -> list[dict]:
       sec = _section(body, "Risks")
       out = []
       for line in sec.splitlines():
           line = line.strip()
           if not line.startswith("|"):
               continue
           cells = [c.strip() for c in line.strip("|").split("|")]
           if len(cells) != 6 or cells[0] in ("id", "----") or set(cells[0]) <= {"-"}:
               continue
           out.append({
               "id": cells[0], "risk": cells[1], "likelihood": cells[2],
               "impact": cells[3], "mitigation": cells[4], "status": cells[5],
           })
       return out


   def parse_edges(body: str) -> list[dict]:
       sec = _section(body, "Edges")
       out = []
       for src, dst, label in re.findall(
           r"-\s*(\w+)\s*->\s*(\w+)\s*\((.*?)\)", sec
       ):
           out.append({"src": src, "dst": dst, "label": label.strip()})
       return out


   def parse_versions(body: str) -> list[dict]:
       sec = _section(body, "Version Log")
       out = []
       for line in sec.splitlines():
           line = line.strip()
           if not line.startswith("- "):
               continue
           parts = [p.strip() for p in line[2:].split("—")]
           if len(parts) >= 3:
               out.append({"version": parts[0], "date": parts[1],
                           "note": "—".join(parts[2:]).strip()})
       return out


   def parse_blueprint(text: str) -> dict:
       """Full parse: markdown -> canonical blueprint dict (see T1.2.03 shape)."""
       fm, body = split_frontmatter(text)
       data = yaml.safe_load(fm)
       if not isinstance(data, dict):
           raise ValueError("frontmatter is not a mapping")
       bp = new_blueprint_dict()
       bp["frontmatter"] = data
       bp["summary"] = parse_summary(body)
       bp["categories"] = parse_categories(body)
       bp["nodes"] = parse_nodes(body)
       bp["assumptions"] = parse_assumptions(body)
       bp["decisions"] = parse_decisions(body)
       bp["risks"] = parse_risks(body)
       bp["research"] = parse_research(body)
       bp["chronicle_links"] = parse_chronicle_links(body)
       bp["edges"] = parse_edges(body)
       bp["versions"] = parse_versions(body)
       return bp
   ```
