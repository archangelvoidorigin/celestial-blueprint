   ```python
   """Round-trip: parse -> serialize -> parse must be stable for all tiers."""
   from pathlib import Path

   import pytest

   from app.services.parser import parse_blueprint
   from app.services.serializer import serialize_blueprint

   TEMPLATES = Path(__file__).resolve().parents[2] / "docs" / "templates"
   TIERS = ["small", "medium", "large", "continuous"]


   def _load(tier: str) -> str:
       return (TEMPLATES / f"{tier}.md").read_text()


   @pytest.mark.parametrize("tier", TIERS)
   def test_roundtrip_stable(tier):
       once = parse_blueprint(_load(tier))
       twice = parse_blueprint(serialize_blueprint(once))
       assert twice == once


   @pytest.mark.parametrize("tier", TIERS)
   def test_frontmatter_preserved(tier):
       bp = parse_blueprint(_load(tier))
       md = serialize_blueprint(bp)
       again = parse_blueprint(md)
       assert again["frontmatter"]["tier"] == tier
       assert again["frontmatter"]["id"] == bp["frontmatter"]["id"]


   @pytest.mark.parametrize("tier", TIERS)
   def test_nodes_count_preserved(tier):
       bp = parse_blueprint(_load(tier))
       again = parse_blueprint(serialize_blueprint(bp))
       assert len(again["nodes"]) == len(bp["nodes"])
       assert len(again["edges"]) == len(bp["edges"])
   ```
