## Context
The parser (T1.2.03–05) turns markdown into the canonical blueprint dict. The serializer does the reverse: blueprint dict → markdown, in exactly the format `docs/FORMAT.md` defines and the parser reads. This is what mutating API endpoints call to rewrite the `.md` file on disk, and what `GET /blueprints/{id}/export` returns. The round-trip contract is: `parse_blueprint(serialize_blueprint(bp))` equals `bp` (verified in T1.2.07). Equality is on the re-parsed dict, not byte-identical text, so formatting need only be parser-readable.

## Objective
Implement `serialize_blueprint(bp: dict) -> str` producing markdown that the parser reads back to an equal dict.

## Instructions
1. Create `backend/app/services/serializer.py` with exactly:
   ```python
   """Blueprint dict -> markdown. Inverse of parser.parse_blueprint."""
   from __future__ import annotations

   import yaml


   def _q(value) -> str:
       """Quote an anchor value if it contains spaces or is empty."""
       s = str(value)
       return f'"{s}"' if (s == "" or " " in s) else s


   def _emit_nodes(nodes: list[dict]) -> str:
       out = ["## Nodes", ""]
       for n in nodes:
           extra = " ".join(f"{k}={_q(v)}" for k, v in n.get("data", {}).items())
           anchor = f"<!-- node:{n['type']} id={n['id']} cat={n['cat']}"
           if extra:
               anchor += f" {extra}"
           anchor += " -->"
           out.append(f"### {n['title']}")
           out.append(anchor)
           if n.get("body"):
               out.append(n["body"])
           out.append("")
       return "\n".join(out)


   def serialize_blueprint(bp: dict) -> str:
       fm = bp.get("frontmatter", {})
       parts: list[str] = []

       # Frontmatter
       fm_yaml = yaml.safe_dump(fm, sort_keys=False, allow_unicode=True).rstrip()
       parts.append(f"---\n{fm_yaml}\n---")

       # Title + summary
       name = fm.get("name") or fm.get("title") or "Blueprint"
       parts.append(f"\n# {name}\n")
       parts.append("<!-- bp:summary -->")
       parts.append(bp.get("summary", ""))

       # Categories
       if bp.get("categories"):
           parts.append("\n## Categories")
           for c in bp["categories"]:
               parts.append(f"- <!-- cat:{c['id']} --> {c['name']}")

       # Nodes
       if bp.get("nodes"):
           parts.append("\n" + _emit_nodes(bp["nodes"]).rstrip())

       # Assumptions
       if bp.get("assumptions"):
           parts.append("\n## Assumptions")
           for a in bp["assumptions"]:
               parts.append(
                   f"- <!-- assumption:{a['id']} bp_ref={a['bp_ref']} "
                   f"status={a['status']} risk={a['risk']} -->"
               )
               if a.get("body"):
                   parts.append(f"  {a['body']}")

       # Decisions
       if bp.get("decisions"):
           parts.append("\n## Decisions")
           for d in bp["decisions"]:
               parts.append(
                   f"- <!-- decision:{d['id']} bp_ref={d['bp_ref']} "
                   f"status={d['status']} date={d['date']} -->"
               )
               if d.get("body"):
                   parts.append(f"  {d['body']}")

       # Risks
       if bp.get("risks"):
           parts.append("\n## Risks\n")
           parts.append("| id | risk | likelihood | impact | mitigation | status |")
           parts.append("|----|------|-----------|--------|-----------|--------|")
           for r in bp["risks"]:
               parts.append(
                   f"| {r['id']} | {r['risk']} | {r['likelihood']} | "
                   f"{r['impact']} | {r['mitigation']} | {r['status']} |"
               )
           parts.append("<!-- risk-table -->")

       # Research Log
       if bp.get("research"):
           parts.append("\n## Research Log")
           for rs in bp["research"]:
               parts.append(
                   f"- <!-- research:{rs['id']} query={_q(rs['query'])} "
                   f"date={rs['date']} -->"
               )
               if rs.get("body"):
                   parts.append(f"  {rs['body']}")

       # Chronicle Links
       if bp.get("chronicle_links"):
           parts.append("\n## Chronicle Links")
           for ch in bp["chronicle_links"]:
               parts.append(
                   f"- <!-- chronicle:{ch['id']} type={ch['type']} --> {ch['path']}"
               )

       # Edges
       if bp.get("edges"):
           parts.append("\n## Edges")
           parts.append("<!-- edges -->")
           for e in bp["edges"]:
               parts.append(f"- {e['src']} -> {e['dst']} ({e['label']})")

       # Version Log
       if bp.get("versions"):
           parts.append("\n## Version Log")
           parts.append("<!-- versions -->")
           for v in bp["versions"]:
               parts.append(f"- {v['version']} — {v['date']} — {v['note']}")

       return "\n".join(parts) + "\n"
   ```

> The Version Log separator is the em-dash `—` (U+2014). Copy it exactly so the parser reads it back.
