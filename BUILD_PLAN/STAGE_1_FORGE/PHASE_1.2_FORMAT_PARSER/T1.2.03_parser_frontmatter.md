## Context
Blueprints are markdown with YAML frontmatter then a body of sections carrying HTML-comment anchors (see `docs/FORMAT.md`). The parser turns a markdown string into one canonical **blueprint dict** that the rest of the backend uses. This is the FIRST parser task: it sets up the module, the frontmatter reader, and a shared anchor-attribute helper that later parser tasks (T1.2.04, T1.2.05) reuse. PyYAML is already installed (T1.1.02).

**Canonical blueprint dict shape** (all later parser tasks fill more of this; serializer T1.2.06 consumes it):
```python
{
  "frontmatter": {},      # dict from YAML frontmatter
  "summary": "",          # prose after <!-- bp:summary -->
  "categories": [],       # [{"id","name"}]
  "nodes": [],            # [{"id","cat","type","title","body","data"{}}]
  "assumptions": [],      # [{"id","bp_ref","status","risk","body"}]
  "decisions": [],        # [{"id","bp_ref","status","date","body"}]
  "risks": [],            # [{"id","risk","likelihood","impact","mitigation","status"}]
  "research": [],         # [{"id","query","date","body"}]
  "chronicle_links": [],  # [{"id","type","path"}]
  "edges": [],            # [{"src","dst","label"}]
  "versions": [],         # [{"version","date","note"}]
}
```

## Instructions
1. Create `backend/app/services/parser.py` with exactly:
   ```python
   """Blueprint markdown parser. Markdown is canonical (see docs/FORMAT.md)."""
   from __future__ import annotations

   import re
   import yaml

   # Matches key=value pairs in an anchor. Value is either a single/double
   # quoted string (may contain spaces) or a run of non-space characters.
   _ATTR_RE = re.compile(r"""(\w+)=('[^']*'|"[^"]*"|\S+)""")


   def split_frontmatter(text: str) -> tuple[str, str]:
       """Return (frontmatter_yaml, body). Raises ValueError if no frontmatter."""
       if not text.startswith("---"):
           raise ValueError("no frontmatter: file must start with '---'")
       parts = text.split("\n", 1)
       rest = parts[1] if len(parts) > 1 else ""
       end = rest.find("\n---")
       if end == -1:
           raise ValueError("no frontmatter: missing closing '---'")
       fm = rest[:end]
       body = rest[end + 4:]  # skip the closing '\n---'
       if body.startswith("\n"):
           body = body[1:]
       return fm, body


   def parse_frontmatter(text: str) -> dict:
       """Parse the YAML frontmatter block into a dict."""
       fm, _ = split_frontmatter(text)
       data = yaml.safe_load(fm)
       if not isinstance(data, dict):
           raise ValueError("frontmatter is not a mapping")
       return data


   def parse_anchor_attrs(anchor: str) -> dict:
       """Parse 'key=value key2="quoted value"' into a dict of strings.

       Quotes are stripped. Used for every HTML-comment anchor in the body.
       """
       attrs: dict[str, str] = {}
       for key, raw in _ATTR_RE.findall(anchor):
           if (raw.startswith("'") and raw.endswith("'")) or (
               raw.startswith('"') and raw.endswith('"')
           ):
               raw = raw[1:-1]
           attrs[key] = raw
       return attrs


   def new_blueprint_dict() -> dict:
       """An empty canonical blueprint dict with all keys present."""
       return {
           "frontmatter": {},
           "summary": "",
           "categories": [],
           "nodes": [],
           "assumptions": [],
           "decisions": [],
           "risks": [],
           "research": [],
           "chronicle_links": [],
           "edges": [],
           "versions": [],
       }
   ```
