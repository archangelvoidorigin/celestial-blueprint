## Context
The NodeCreator is a compact sidebar panel on the canvas that lets the user add a new node to the current blueprint. It calls `POST /api/blueprints/:id/nodes` (via `addNode` from the api client) and refreshes the React Flow graph. Fields: title (required), type (dropdown of common types), optional body. The `blueprintId` and a callback `onAdded(updatedBp)` are passed as props.

## Objective
Create `NodeCreator.tsx` and wire it into `Canvas.tsx`.

## Instructions
1. Create `frontend/src/components/canvas/NodeCreator.tsx` with exactly:
   ```tsx
   import React, { useState } from "react";
   import { addNode } from "../../lib/api";
   import type { Blueprint, NodeType } from "../../types/blueprint";

   const NODE_TYPES: NodeType[] = [
     "text", "metric", "assumption", "decision",
     "research", "chronicle", "agent", "cost", "tool",
   ];

   interface Props {
     blueprintId: string;
     onAdded: (bp: Blueprint) => void;
   }

   export default function NodeCreator({ blueprintId, onAdded }: Props) {
     const [open, setOpen] = useState(false);
     const [title, setTitle] = useState("");
     const [type, setType] = useState<NodeType>("text");
     const [body, setBody] = useState("");

     async function handleSubmit(e: React.FormEvent) {
       e.preventDefault();
       if (!title.trim()) return;
       const id = `n${Date.now()}`;
       const updated = await addNode(blueprintId, { id, type, title: title.trim(), body });
       onAdded(updated);
       setTitle(""); setBody(""); setOpen(false);
     }

     return (
       <div className="absolute right-4 top-14 z-10">
         {!open ? (
           <button
             className="rounded bg-accent px-3 py-1.5 text-sm font-medium text-white shadow hover:bg-violet-500"
             onClick={() => setOpen(true)}
           >
             + Node
           </button>
         ) : (
           <form
             onSubmit={handleSubmit}
             className="w-56 rounded-lg border border-border bg-surface p-3 shadow-lg"
           >
             <p className="mb-2 text-xs font-semibold uppercase text-muted">Add node</p>
             <input
               className="mb-2 w-full rounded border border-border bg-bg px-2 py-1 text-sm text-ink placeholder-muted"
               placeholder="Title"
               value={title}
               onChange={(e) => setTitle(e.target.value)}
               autoFocus
             />
             <select
               className="mb-2 w-full rounded border border-border bg-bg px-2 py-1 text-sm text-ink"
               value={type}
               onChange={(e) => setType(e.target.value as NodeType)}
             >
               {NODE_TYPES.map((t) => <option key={t}>{t}</option>)}
             </select>
             <textarea
               className="mb-2 w-full rounded border border-border bg-bg px-2 py-1 text-sm text-ink placeholder-muted"
               placeholder="Body (optional)"
               rows={2}
               value={body}
               onChange={(e) => setBody(e.target.value)}
             />
             <div className="flex gap-2">
               <button
                 type="submit"
                 className="flex-1 rounded bg-accent py-1 text-xs text-white hover:bg-violet-500"
               >
                 Add
               </button>
               <button
                 type="button"
                 className="flex-1 rounded border border-border py-1 text-xs text-muted hover:text-ink"
                 onClick={() => setOpen(false)}
               >
                 Cancel
               </button>
             </div>
           </form>
         )}
       </div>
     );
   }
   ```
