## Context
Chronicles are the session journal. The ChroniclePanel is a collapsible bottom drawer on the canvas that lists existing chronicles and lets the user write a new RAW entry without leaving the canvas. It calls `listChronicles` and `addChronicle` from the API client.

## Objective
Create `ChroniclePanel.tsx` with list + add, wired into `Canvas.tsx` as a bottom drawer.

## Instructions
1. Create `frontend/src/components/chronicle/ChroniclePanel.tsx` with exactly:
   ```tsx
   import React, { useEffect, useState } from "react";
   import { listChronicles, addChronicle } from "../../lib/api";
   import type { Chronicle } from "../../types/blueprint";

   interface Props { blueprintId: string; }

   export default function ChroniclePanel({ blueprintId }: Props) {
     const [open, setOpen] = useState(false);
     const [chronicles, setChronicles] = useState<Chronicle[]>([]);
     const [body, setBody] = useState("");
     const [trigger, setTrigger] = useState("");

     useEffect(() => {
       if (!open) return;
       listChronicles(blueprintId).then(setChronicles).catch(console.error);
     }, [open, blueprintId]);

     async function handleAdd(e: React.FormEvent) {
       e.preventDefault();
       if (!body.trim()) return;
       await addChronicle(blueprintId, { kind: "raw", trigger, body });
       const updated = await listChronicles(blueprintId);
       setChronicles(updated);
       setBody(""); setTrigger("");
     }

     return (
       <div className="border-t border-border bg-surface">
         <button
           className="w-full px-4 py-2 text-left text-xs font-semibold uppercase text-muted hover:text-ink"
           onClick={() => setOpen((v) => !v)}
         >
           {open ? "▾" : "▸"} Chronicle log ({chronicles.length})
         </button>

         {open && (
           <div className="max-h-64 overflow-y-auto px-4 pb-4">
             <form onSubmit={handleAdd} className="mb-3 flex gap-2">
               <input
                 className="w-32 rounded border border-border bg-bg px-2 py-1 text-xs text-ink placeholder-muted"
                 placeholder="trigger (optional)"
                 value={trigger}
                 onChange={(e) => setTrigger(e.target.value)}
               />
               <input
                 className="flex-1 rounded border border-border bg-bg px-2 py-1 text-xs text-ink placeholder-muted"
                 placeholder="Write a RAW chronicle note…"
                 value={body}
                 onChange={(e) => setBody(e.target.value)}
               />
               <button
                 type="submit"
                 className="rounded bg-accent px-3 py-1 text-xs text-white hover:bg-violet-500"
               >
                 Log
               </button>
             </form>

             {chronicles.length === 0 ? (
               <p className="text-xs text-muted">No chronicles yet.</p>
             ) : (
               <ul className="space-y-2">
                 {chronicles.map((c) => (
                   <li key={c.id} className="rounded border border-border px-3 py-2 text-xs">
                     <span className="mr-2 font-semibold text-accent">{c.kind}</span>
                     {c.trigger && <span className="mr-2 text-muted">{c.trigger}</span>}
                     <span className="text-ink">{c.body.slice(0, 120)}{c.body.length > 120 ? "…" : ""}</span>
                   </li>
                 ))}
               </ul>
             )}
           </div>
         )}
       </div>
     );
   }
   ```
2. Edit `frontend/src/pages/Canvas.tsx`:
