## Context
The list page is the home screen — it shows all blueprints and lets the user create a new one. It uses the `listBlueprints` and `createBlueprint` API calls, and navigates to the canvas page (T1.6.04) on click. The canvas page URL pattern is `/blueprint/:id`. React Router (already in `package.json`) handles routing; it's wired in T1.6.08.

## Objective
Create `BlueprintList.tsx` with list, search filter, and new-blueprint creation.

## Instructions
1. Create `frontend/src/pages/BlueprintList.tsx` with exactly:
   ```tsx
   import React, { useEffect, useState } from "react";
   import { useNavigate } from "react-router-dom";
   import { listBlueprints, createBlueprint } from "../lib/api";
   import type { BlueprintSummary } from "../types/blueprint";

   const TIER_COLOR: Record<string, string> = {
     small: "text-emerald-400",
     medium: "text-blue-400",
     large: "text-violet-400",
     continuous: "text-amber-400",
   };

   const STATUS_COLOR: Record<string, string> = {
     draft: "text-muted",
     active: "text-emerald-400",
     blocked: "text-red-400",
     shipped: "text-blue-400",
     archived: "text-muted line-through",
   };

   export default function BlueprintList() {
     const nav = useNavigate();
     const [blueprints, setBlueprints] = useState<BlueprintSummary[]>([]);
     const [q, setQ] = useState("");
     const [creating, setCreating] = useState(false);
     const [newName, setNewName] = useState("");
     const [newTier, setNewTier] = useState<string>("small");

     useEffect(() => {
       listBlueprints().then(setBlueprints).catch(console.error);
     }, []);

     const filtered = blueprints.filter((b) =>
       b.name.toLowerCase().includes(q.toLowerCase())
     );

     async function handleCreate(e: React.FormEvent) {
       e.preventDefault();
       if (!newName.trim()) return;
       const bp = await createBlueprint({ name: newName.trim(), tier: newTier });
       const id = (bp.frontmatter as Record<string, string>).id;
       nav(`/blueprint/${id}`);
     }

     return (
       <div className="mx-auto max-w-4xl px-4 py-8">
         <div className="mb-6 flex items-center justify-between">
           <h1 className="text-2xl font-semibold text-ink">Blueprints</h1>
           <button
             className="rounded bg-accent px-4 py-1.5 text-sm font-medium text-white hover:bg-violet-500"
             onClick={() => setCreating((v) => !v)}
           >
             + New
           </button>
         </div>

         {creating && (
           <form
             onSubmit={handleCreate}
             className="mb-6 flex gap-2 rounded-lg border border-border bg-surface p-4"
           >
             <input
               className="flex-1 rounded border border-border bg-bg px-3 py-1.5 text-sm text-ink placeholder-muted"
               placeholder="Blueprint name"
               value={newName}
               onChange={(e) => setNewName(e.target.value)}
               autoFocus
             />
             <select
               className="rounded border border-border bg-bg px-2 py-1.5 text-sm text-ink"
               value={newTier}
               onChange={(e) => setNewTier(e.target.value)}
             >
               {["small", "medium", "large", "continuous"].map((t) => (
                 <option key={t}>{t}</option>
               ))}
             </select>
             <button
               type="submit"
               className="rounded bg-accent px-4 py-1.5 text-sm text-white hover:bg-violet-500"
             >
               Create
             </button>
           </form>
         )}

         <input
           className="mb-4 w-full rounded border border-border bg-bg px-3 py-1.5 text-sm text-ink placeholder-muted"
           placeholder="Search blueprints…"
           value={q}
           onChange={(e) => setQ(e.target.value)}
         />

         {filtered.length === 0 ? (
           <p className="text-sm text-muted">No blueprints yet.</p>
         ) : (
           <ul className="space-y-2">
             {filtered.map((b) => (
               <li
                 key={b.id}
                 className="flex cursor-pointer items-center justify-between rounded-lg border border-border bg-surface px-4 py-3 hover:border-accent"
                 onClick={() => nav(`/blueprint/${b.id}`)}
               >
                 <div>
                   <span className="font-medium text-ink">{b.name}</span>
                   <span className="ml-2 text-xs text-muted">v{b.version}</span>
                 </div>
                 <div className="flex gap-3 text-xs">
                   <span className={TIER_COLOR[b.tier] ?? "text-muted"}>{b.tier}</span>
                   <span className={STATUS_COLOR[b.status] ?? "text-muted"}>{b.status}</span>
                 </div>
               </li>
             ))}
           </ul>
         )}
       </div>
     );
   }
   ```
