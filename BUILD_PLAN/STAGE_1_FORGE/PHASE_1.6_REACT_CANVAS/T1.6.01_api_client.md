   /**
    * Typed API client. Base URL = /api (proxied to :8787 in dev).
    * All functions throw on non-2xx.
    */

   const BASE = "/api";

   async function req<T>(
     method: string,
     path: string,
     body?: unknown
   ): Promise<T> {
     const r = await fetch(`${BASE}${path}`, {
       method,
       headers: { "Content-Type": "application/json" },
       body: body !== undefined ? JSON.stringify(body) : undefined,
     });
     if (!r.ok) {
       const msg = await r.text().catch(() => r.statusText);
       throw new Error(`${method} ${path} → ${r.status}: ${msg}`);
     }
     return r.json() as Promise<T>;
   }

   // Health
   export const health = () => req<{ status: string }>("GET", "/health");

   // Blueprints
   export const listBlueprints = () =>
     req<BlueprintSummary[]>("GET", "/blueprints");

   export const getBlueprint = (id: string) =>
     req<Blueprint>("GET", `/blueprints/${id}`);

   export const searchBlueprints = (q: string) =>
     req<BlueprintSummary[]>("GET", `/blueprints/search?q=${encodeURIComponent(q)}`);

   export const createBlueprint = (data: {
     name: string;
     tier?: string;
     realm?: string;
     summary?: string;
   }) => req<Blueprint>("POST", "/blueprints", data);

   export const patchBlueprint = (
     id: string,
     data: { name?: string; status?: string; version?: string; summary?: string }
   ) => req<Blueprint>("PATCH", `/blueprints/${id}`, data);

   export const archiveBlueprint = (id: string) =>
     req<{ status: string; id: string }>("POST", `/blueprints/${id}/archive`);

   // Nodes & Edges
   export const addNode = (id: string, node: Partial<Node>) =>
     req<Blueprint>("POST", `/blueprints/${id}/nodes`, node);

   export const addEdge = (id: string, edge: { src: string; dst: string; label?: string }) =>
     req<Blueprint>("POST", `/blueprints/${id}/edges`, edge);

   // Tracking
   export const addAssumption = (id: string, a: Partial<Assumption>) =>
     req<Blueprint>("POST", `/blueprints/${id}/assumptions`, a);

   export const addDecision = (id: string, d: Partial<Decision>) =>
     req<Blueprint>("POST", `/blueprints/${id}/decisions`, d);

   export const addRisk = (id: string, r: Partial<Risk>) =>
     req<Blueprint>("POST", `/blueprints/${id}/risks`, r);

   // Chronicles
   export const addChronicle = (
     id: string,
     c: { kind?: string; trigger?: string; cycle?: string; body?: string }
   ) => req<{ id: number; path: string }>("POST", `/blueprints/${id}/chronicles`, c);

   export const listChronicles = (id: string) =>
     req<Chronicle[]>("GET", `/blueprints/${id}/chronicles`);

   // I/O
   export const exportBlueprint = (id: string) =>
     fetch(`${BASE}/blueprints/${id}/export`).then((r) =>
       r.ok ? r.text() : Promise.reject(new Error(r.statusText))
     );

   export const importBlueprint = (markdown: string) =>
     req<Blueprint>("POST", "/blueprints/import", { markdown });

   export const rebuild = () =>
     req<{ imported: number }>("POST", "/rebuild");

   // ----- local type stubs (T1.6.02 will formalise these) -----
   export interface BlueprintSummary {
     id: string;
     name: string;
     version: string;
     tier: string;
     status: string;
     updated_at: string;
   }

   export interface Node {
     id: string;
     cat: string;
     type: string;
     title: string;
     body: string;
     data: Record<string, unknown>;
   }

   export interface Edge {
     src: string;
     dst: string;
     label: string;
   }

   export interface Assumption { id: string; bp_ref: string; status: string; risk: string; body: string; }
   export interface Decision { id: string; bp_ref: string; status: string; date: string; body: string; }
   export interface Risk { id: string; risk: string; likelihood: string; impact: string; mitigation: string; status: string; }
   export interface Research { id: string; query: string; date: string; body: string; }
   export interface ChronicleLink { id: string; type: string; path: string; }
   export interface Version { version: string; date: string; note: string; }
   export interface Chronicle { id: number; blueprint_id: string; kind: string; trigger: string; cycle: string; path: string; body: string; created_at: string; }
   export interface Blueprint {
     frontmatter: Record<string, unknown>;
     summary: string;
     categories: { id: string; name: string }[];

     assumptions: Assumption[];
     decisions: Decision[];

     research: Research[];
     chronicle_links: ChronicleLink[];
     edges: Edge[];
     versions: Version[];
   }
