# TASK T1.6.05 — `components/nodes/` node card components

- **Stage / Phase**: Stage 1 — Forge / Phase 1.6
- **Depends on**: T1.6.04
- **Produces**: `frontend/src/components/nodes/TextNode.tsx`, `MetricNode.tsx`, `AssumptionNode.tsx`, `DecisionNode.tsx`, `index.ts`

## Context
React Flow renders custom nodes by passing `type` to a `nodeTypes` map. This task implements four card-style custom node components for the most common node types (`text`, `metric`, `assumption`, `decision`) and an `index.ts` that exports the `nodeTypes` map. The canvas (T1.6.04) imports this map and passes it to `<ReactFlow nodeTypes={...}>`. Node data (title, body, machine fields) comes from the React Flow node's `data.node` (the raw blueprint node set in `bpToRFNodes`).

## Objective
Create four node card components and export them as the `nodeTypes` map.

## Instructions

2. Create `frontend/src/components/nodes/MetricNode.tsx` with exactly:
   ```tsx
   import { Handle, Position } from "reactflow";
   import type { NodeProps } from "reactflow";

   export default function MetricNode({ data }: NodeProps) {
     const d = data.node?.data ?? {};
     return (
       <div className="min-w-[160px] rounded-lg border border-blue-700 bg-surface px-3 py-2 text-sm shadow">

         <p className="font-medium text-blue-300">{data.label}</p>

           {d.value ?? "—"}<span className="ml-1 text-xs text-muted">{d.unit}</span>
   ```

3. Create `frontend/src/components/nodes/AssumptionNode.tsx` with exactly:
   ```tsx
   import { Handle, Position } from "reactflow";
   import type { NodeProps } from "reactflow";

   const RISK_COLOR: Record<string, string> = {
     high: "border-red-600 bg-red-950",
     med: "border-yellow-600 bg-yellow-950",
     low: "border-green-700 bg-green-950",
   };

   export default function AssumptionNode({ data }: NodeProps) {
     const risk = (data.node?.data?.risk as string) ?? "med";
     return (
       <div className={`min-w-[160px] max-w-[240px] rounded-lg border px-3 py-2 text-sm shadow ${RISK_COLOR[risk] ?? "border-border bg-surface"}`}>
         <Handle type="target" position={Position.Top} className="!bg-red-500" />
         <p className="text-xs font-semibold uppercase text-red-400">Assumption</p>
         <p className="mt-0.5 font-medium text-ink">{data.label}</p>
         {data.node?.data?.falsifier && (
           <p className="mt-1 text-xs text-muted">↯ {data.node.data.falsifier}</p>
         )}
         <Handle type="source" position={Position.Bottom} className="!bg-red-500" />
       </div>
     );
   }
   ```
4. Create `frontend/src/components/nodes/DecisionNode.tsx` with exactly:
   ```tsx
   import { Handle, Position } from "reactflow";
   import type { NodeProps } from "reactflow";

   const STATUS_COLOR: Record<string, string> = {
     accepted: "text-emerald-400",
     rejected: "text-red-400",
     deferred: "text-amber-400",

       <div className="min-w-[160px] max-w-[240px] rounded-lg border border-violet-700 bg-surface px-3 py-2 text-sm shadow">

         <p className="mt-0.5 font-medium text-ink">{data.label}</p>
         <p className={`mt-1 text-xs ${STATUS_COLOR[status] ?? "text-muted"}`}>{status}</p>
         <Handle type="source" position={Position.Bottom} className="!bg-violet-500" />
       </div>
     );
   }
   ```

5. Create `frontend/src/components/nodes/index.ts` with exactly:
