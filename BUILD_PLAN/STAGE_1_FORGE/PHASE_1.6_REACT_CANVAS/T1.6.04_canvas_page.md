## Context
The canvas is the main working view: a React Flow graph of a single blueprint's nodes and edges on a dark dot-grid background. It fetches the blueprint from the API, converts blueprint nodes/edges to React Flow format, and renders them. Custom node components (T1.6.05), the NodeCreator sidebar (T1.6.06), and the ChroniclePanel (T1.6.07) slot in as children. The `blueprintId` comes from the React Router param `:id`. reactflow 11.x is already installed.

## Objective
Create `Canvas.tsx` that loads and renders a blueprint as a React Flow graph.

## Instructions
1. Create `frontend/src/pages/Canvas.tsx` with exactly:
   ```tsx
   import React, { useCallback, useEffect, useState } from "react";
   import { useParams, useNavigate } from "react-router-dom";
   import ReactFlow, {
     Background,
     BackgroundVariant,
     Controls,
     MiniMap,
     addEdge,
     useEdgesState,
     useNodesState,
   } from "reactflow";
   import type { Connection, Edge as RFEdge, Node as RFNode } from "reactflow";
   import "reactflow/dist/style.css";

   import { getBlueprint, addEdge as apiAddEdge } from "../lib/api";
   import type { Blueprint } from "../types/blueprint";

   function bpToRFNodes(bp: Blueprint): RFNode[] {
     return bp.nodes.map((n, i) => ({
       id: n.id,
       type: n.type === "metric" ? "metric" : "default",
       position: { x: 60 + (i % 5) * 220, y: 60 + Math.floor(i / 5) * 160 },
       data: { label: n.title || n.id, node: n },
     }));
   }

   function bpToRFEdges(bp: Blueprint): RFEdge[] {
     return bp.edges.map((e, i) => ({
       id: `e-${i}-${e.src}-${e.dst}`,
       source: e.src,
       target: e.dst,
       label: e.label,
       style: { stroke: "#7c5cff" },
     }));
   }

   export default function Canvas() {
     const { id } = useParams<{ id: string }>();
     const nav = useNavigate();
     const [bp, setBp] = useState<Blueprint | null>(null);
     const [nodes, setNodes, onNodesChange] = useNodesState([]);
     const [edges, setEdges, onEdgesChange] = useEdgesState([]);

     useEffect(() => {
       if (!id) return;
       getBlueprint(id).then((b) => {
         setBp(b);
         setNodes(bpToRFNodes(b));
         setEdges(bpToRFEdges(b));
       }).catch(console.error);
     }, [id]);

     const onConnect = useCallback(
       async (params: Connection) => {
         if (!id || !params.source || !params.target) return;
         const updated = await apiAddEdge(id, {
           src: params.source,
           dst: params.target,
           label: "",
         });
         setEdges(bpToRFEdges(updated));
       },
       [id]
     );

     const name = (bp?.frontmatter as Record<string, string> | undefined)?.name ?? id;

     return (
       <div className="flex h-full flex-col">
         {/* Header bar */}
         <div className="flex items-center gap-3 border-b border-border bg-surface px-4 py-2">
           <button
             className="text-sm text-muted hover:text-ink"
             onClick={() => nav("/")}
           >
             ← Blueprints
           </button>
           <span className="text-sm font-medium text-ink">{name}</span>
           {bp && (
             <span className="text-xs text-muted">
               v{(bp.frontmatter as Record<string, string>).version} ·{" "}
               {(bp.frontmatter as Record<string, string>).tier}
             </span>
           )}
         </div>

         {/* Canvas */}
         <div className="flex-1">
           <ReactFlow
             nodes={nodes}
             edges={edges}
             onNodesChange={onNodesChange}
             onEdgesChange={onEdgesChange}
             onConnect={onConnect}
             fitView
           >
             <Background
               variant={BackgroundVariant.Dots}
               gap={20}
               size={1}
               color="#1e2430"
             />
             <Controls />
             <MiniMap
               nodeColor={() => "#7c5cff"}
               maskColor="rgba(11,14,20,0.7)"
             />
           </ReactFlow>
         </div>
       </div>
     );
   }
   ```
