- **Stage / Phase**: Stage 3 — Hands / Phase 3.1

## Context

`cli/main.py` is the `cb` entry point registered in `pyproject.toml` as `cb = "cli.main:main"` (T3.1.01). It builds an argparse tree, calls `get_config()` (T3.1.02) to obtain `api_url` and `token`, constructs a `CelestialClient` (T3.1.03), dispatches to the appropriate method, then prints JSON to stdout. Two output flags: `--pretty` prints indented JSON; `--field NAME` extracts a single top-level key from the JSON response. `blueprint export` is the exception — it prints raw markdown, not JSON. Install the entry point with `pip install -e .` after writing the file.

## Objective

Create `backend/cli/main.py` with the full argparse command tree and `main()` function, then install the `cb` entry point.

## Instructions

1. Create `backend/cli/main.py` with exactly:
   ```python
   """cb — Celestial Blueprint CLI.

   Thin argparse wrapper over the REST API.
   Usage: cb <subcommand> ... [--pretty] [--field NAME]

   The output flags --pretty / --field are accepted after the leaf command
   (e.g. `cb blueprint list --pretty`). They are attached to every leaf
   subparser via a shared parent parser so they work in trailing position.
   """
   from __future__ import annotations

   import argparse
   import json
   import sys

   from cli.client import CelestialClient
   from cli.config import get_config


   # ---------------------------------------------------------------------------
   # output helpers
   # ---------------------------------------------------------------------------

   def _emit(data: object, pretty: bool, field: str | None) -> None:
       """Print JSON (or a single field) to stdout."""
       if field is not None:
           if not isinstance(data, dict):
               print(f"Error: --field requires a JSON object response", file=sys.stderr)
               sys.exit(1)
           if field not in data:
               print(f"Error: field '{field}' not in response", file=sys.stderr)
               sys.exit(1)
           data = data[field]
       indent = 2 if pretty else None
       print(json.dumps(data, indent=indent))


   # ---------------------------------------------------------------------------
   # argument parser
   # ---------------------------------------------------------------------------

   def _build_parser() -> argparse.ArgumentParser:
       # Shared output flags, attached to every leaf subparser via
       # parents=[common] so `--pretty` / `--field` work in trailing position,
       # e.g. `cb blueprint list --pretty`. They are intentionally NOT on the
       # top parser: a leading `cb --pretty blueprint list` would be silently
       # overridden by the leaf's default, so it is rejected loudly instead.
       common = argparse.ArgumentParser(add_help=False)
       common.add_argument("--pretty", action="store_true", help="Pretty-print JSON output")
       common.add_argument("--field", metavar="NAME", help="Extract one top-level field from JSON")

       p = argparse.ArgumentParser(
           prog="cb",
           description="Celestial Blueprint CLI — wraps the REST API",
       )

       sub = p.add_subparsers(dest="group", required=True)

       # ---- blueprint -------------------------------------------------------
       bp = sub.add_parser("blueprint", help="Blueprint commands")
       bp_sub = bp.add_subparsers(dest="action", required=True)

       bp_sub.add_parser("list", parents=[common], help="List all blueprints")

       bp_get = bp_sub.add_parser("get", parents=[common], help="Get a blueprint by ID")
       bp_get.add_argument("id", help="Blueprint ID")

       bp_create = bp_sub.add_parser("create", parents=[common], help="Create a blueprint")
       bp_create.add_argument("name", help="Blueprint name")
       bp_create.add_argument("--tier", choices=["small", "medium", "large", "continuous"])
       bp_create.add_argument("--summary", metavar="TEXT")

       bp_patch = bp_sub.add_parser("patch", parents=[common], help="Patch a blueprint")
       bp_patch.add_argument("id", help="Blueprint ID")
       bp_patch.add_argument("--status", metavar="STATUS")
       bp_patch.add_argument("--name", metavar="NAME")
       bp_patch.add_argument("--version", metavar="VER")
       bp_patch.add_argument("--summary", metavar="TEXT")

       bp_archive = bp_sub.add_parser("archive", parents=[common], help="Archive a blueprint")
       bp_archive.add_argument("id", help="Blueprint ID")

       bp_export = bp_sub.add_parser("export", parents=[common], help="Export blueprint as markdown")
       bp_export.add_argument("id", help="Blueprint ID")

       bp_import = bp_sub.add_parser("import", parents=[common], help="Import blueprint from markdown file")
       bp_import.add_argument("file", help="Path to .md file")

       # ---- node ------------------------------------------------------------
       node = sub.add_parser("node", help="Node commands")
       node_sub = node.add_subparsers(dest="action", required=True)

       node_add = node_sub.add_parser("add", parents=[common], help="Add a node to a blueprint")
       node_add.add_argument("blueprint_id", help="Blueprint ID")
       node_add.add_argument("--type", dest="node_type", required=True, metavar="TYPE")
       node_add.add_argument("--title", required=True, metavar="TITLE")
       node_add.add_argument("--cat", metavar="CAT")
       node_add.add_argument("--id", dest="node_id", metavar="ID")
       node_add.add_argument("--body", metavar="TEXT")

       # ---- edge ------------------------------------------------------------
       edge = sub.add_parser("edge", help="Edge commands")
       edge_sub = edge.add_subparsers(dest="action", required=True)

       edge_add = edge_sub.add_parser("add", parents=[common], help="Add an edge to a blueprint")
       edge_add.add_argument("blueprint_id", help="Blueprint ID")
       edge_add.add_argument("--src", required=True, metavar="SRC")
       edge_add.add_argument("--dst", required=True, metavar="DST")
       edge_add.add_argument("--label", metavar="LABEL")

       # ---- assumption ------------------------------------------------------
       assumption = sub.add_parser("assumption", help="Assumption commands")
       assumption_sub = assumption.add_subparsers(dest="action", required=True)

       assumption_add = assumption_sub.add_parser("add", parents=[common], help="Add an assumption")
       assumption_add.add_argument("blueprint_id", help="Blueprint ID")
       assumption_add.add_argument("--id", dest="assumption_id", required=True, metavar="ID")
       assumption_add.add_argument("--bp-ref", metavar="REF")
       assumption_add.add_argument("--status", metavar="STATUS")
       assumption_add.add_argument("--risk", metavar="RISK")
       assumption_add.add_argument("--body", metavar="TEXT")

       # ---- decision --------------------------------------------------------
       decision = sub.add_parser("decision", help="Decision commands")
       decision_sub = decision.add_subparsers(dest="action", required=True)

       decision_add = decision_sub.add_parser("add", parents=[common], help="Add a decision")
       decision_add.add_argument("blueprint_id", help="Blueprint ID")
       decision_add.add_argument("--id", dest="decision_id", required=True, metavar="ID")
       decision_add.add_argument("--bp-ref", metavar="REF")
       decision_add.add_argument("--status", metavar="STATUS")
       decision_add.add_argument("--date", metavar="DATE")
       decision_add.add_argument("--body", metavar="TEXT")

       # ---- risk ------------------------------------------------------------
       risk = sub.add_parser("risk", help="Risk commands")
       risk_sub = risk.add_subparsers(dest="action", required=True)

       risk_add = risk_sub.add_parser("add", parents=[common], help="Add a risk")
       risk_add.add_argument("blueprint_id", help="Blueprint ID")
       risk_add.add_argument("--id", dest="risk_id", required=True, metavar="ID")
       risk_add.add_argument("--risk", dest="risk_text", metavar="TEXT")
       risk_add.add_argument("--likelihood", metavar="L")
       risk_add.add_argument("--impact", metavar="I")
       risk_add.add_argument("--mitigation", metavar="M")
       risk_add.add_argument("--status", metavar="S")

       # ---- chronicle -------------------------------------------------------
       chronicle = sub.add_parser("chronicle", help="Chronicle commands")
       chronicle_sub = chronicle.add_subparsers(dest="action", required=True)

       chronicle_add = chronicle_sub.add_parser("add", parents=[common], help="Add a chronicle entry")
       chronicle_add.add_argument("blueprint_id", help="Blueprint ID")
       chronicle_add.add_argument("--kind", choices=["raw", "cycle"], default="raw")
       chronicle_add.add_argument("--trigger", metavar="TEXT")
       chronicle_add.add_argument("--body", metavar="TEXT")

       chronicle_list = chronicle_sub.add_parser("list", parents=[common], help="List chronicle entries")
       chronicle_list.add_argument("blueprint_id", help="Blueprint ID")

       # ---- rebuild ---------------------------------------------------------
       sub.add_parser("rebuild", parents=[common], help="Rebuild DB from markdown files")

       return p


   # ---------------------------------------------------------------------------
   # dispatch
   # ---------------------------------------------------------------------------

   def main() -> None:
       parser = _build_parser()
       args = parser.parse_args()

       cfg = get_config()
       client = CelestialClient(api_url=cfg["api_url"], token=cfg["token"])

       pretty: bool = args.pretty
       field: str | None = args.field

       try:
           group: str = args.group

           if group == "blueprint":
               action: str = args.action
               if action == "list":
                   _emit(client.list_blueprints(), pretty, field)
               elif action == "get":
                   _emit(client.get_blueprint(args.id), pretty, field)
               elif action == "create":
                   _emit(
                       client.create_blueprint(
                           name=args.name,
                           tier=args.tier,
                           summary=args.summary,
                       ),
                       pretty,
                       field,
                   )
               elif action == "patch":
                   _emit(
                       client.patch_blueprint(
                           bp_id=args.id,
                           name=args.name,
                           status=args.status,
                           version=args.version,
                           summary=args.summary,
                       ),
                       pretty,
                       field,
                   )
               elif action == "archive":
                   _emit(client.archive_blueprint(args.id), pretty, field)
               elif action == "export":
                   # Raw markdown — skip JSON encoding
                   print(client.export_blueprint(args.id))
               elif action == "import":
                   with open(args.file, "r", encoding="utf-8") as fh:
                       markdown = fh.read()
                   _emit(client.import_blueprint(markdown), pretty, field)

           elif group == "node":
               if args.action == "add":
                   _emit(
                       client.add_node(
                           bp_id=args.blueprint_id,
                           node_type=args.node_type,
                           title=args.title,
                           cat=args.cat,
                           node_id=args.node_id,
                           body=args.body,
                       ),
                       pretty,
                       field,
                   )

           elif group == "edge":
               if args.action == "add":
                   _emit(
                       client.add_edge(
                           bp_id=args.blueprint_id,
                           src=args.src,
                           dst=args.dst,
                           label=args.label,
                       ),
                       pretty,
                       field,
                   )

           elif group == "assumption":
               if args.action == "add":
                   _emit(
                       client.add_assumption(
                           bp_id=args.blueprint_id,
                           assumption_id=args.assumption_id,
                           bp_ref=args.bp_ref,
                           status=args.status,
                           risk=args.risk,
                           body=args.body,
                       ),
                       pretty,
                       field,
                   )

           elif group == "decision":
               if args.action == "add":
                   _emit(
                       client.add_decision(
                           bp_id=args.blueprint_id,
                           decision_id=args.decision_id,
                           bp_ref=args.bp_ref,
                           status=args.status,
                           date=args.date,
                           body=args.body,
                       ),
                       pretty,
                       field,
                   )

           elif group == "risk":
               if args.action == "add":
                   _emit(
                       client.add_risk(
                           bp_id=args.blueprint_id,
                           risk_id=args.risk_id,
                           risk=args.risk_text,
                           likelihood=args.likelihood,
                           impact=args.impact,
                           mitigation=args.mitigation,
                           status=args.status,
                       ),
                       pretty,
                       field,
                   )

           elif group == "chronicle":
               if args.action == "add":
                   _emit(
                       client.add_chronicle(
                           bp_id=args.blueprint_id,
                           kind=args.kind,
                           trigger=args.trigger,
                           body=args.body,
                       ),
                       pretty,
                       field,
                   )
               elif args.action == "list":
                   _emit(client.list_chronicles(args.blueprint_id), pretty, field)

           elif group == "rebuild":
               _emit(client.rebuild(), pretty, field)

       finally:
           client.close()


   if __name__ == "__main__":
       main()
   ```
