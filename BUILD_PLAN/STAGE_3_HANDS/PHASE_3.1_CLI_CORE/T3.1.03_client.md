## Context

The `cb` CLI is a pure HTTP client — it never imports `app.*`. All API calls go through a single `CelestialClient` class backed by `httpx.Client`. On any non-2xx response the client prints the error to stderr and calls `sys.exit(1)`. The `export_blueprint` method returns raw markdown text (the server sends `text/plain`). All other methods return the parsed JSON response dict/list.

The API (built in Stage 1) runs on `http://localhost:8787` by default. Auth header `Authorization: Bearer <token>` is added only when `token` is non-empty.

## Objective

Create `backend/cli/client.py` with `CelestialClient` covering every API endpoint.

   ```python
   """HTTP client wrapping the Celestial Blueprint REST API.

   Never imports app.*  — pure httpx over the wire.
   """
   from __future__ import annotations

   import sys
   from typing import Any

   import httpx


   class CelestialClient:
       def __init__(self, api_url: str, token: str = "") -> None:
           headers: dict[str, str] = {}
           if token:
               headers["Authorization"] = f"Bearer {token}"
           self._client = httpx.Client(base_url=api_url, headers=headers, timeout=30)

       # ------------------------------------------------------------------
       # internal helpers
       # ------------------------------------------------------------------

       def _get(self, path: str) -> Any:
           return self._ok(self._client.get(path))

       def _get_text(self, path: str) -> str:
           r = self._client.get(path)
           if r.status_code >= 300:
               print(f"Error {r.status_code}: {r.text}", file=sys.stderr)
               sys.exit(1)
           return r.text

       def _post(self, path: str, body: dict | None = None) -> Any:
           return self._ok(self._client.post(path, json=body or {}))

       def _patch(self, path: str, body: dict) -> Any:
           return self._ok(self._client.patch(path, json=body))

       @staticmethod
       def _ok(r: httpx.Response) -> Any:
           if r.status_code >= 300:
               print(f"Error {r.status_code}: {r.text}", file=sys.stderr)
               sys.exit(1)
           return r.json()

       # ------------------------------------------------------------------
       # health
       # ------------------------------------------------------------------

       def health(self) -> dict:
           return self._get("/api/health")

       # ------------------------------------------------------------------
       # blueprints
       # ------------------------------------------------------------------

       def list_blueprints(self) -> list:
           return self._get("/api/blueprints")

       def search_blueprints(self, q: str) -> list:
           return self._ok(self._client.get("/api/blueprints/search", params={"q": q}))

       def get_blueprint(self, bp_id: str) -> dict:
           return self._get(f"/api/blueprints/{bp_id}")

       def create_blueprint(
           self,
           name: str,
           tier: str | None = None,
           realm: str | None = None,
           summary: str | None = None,
       ) -> dict:
           body: dict[str, Any] = {"name": name}
           if tier is not None:
               body["tier"] = tier
           if realm is not None:
               body["realm"] = realm
           if summary is not None:
               body["summary"] = summary
           return self._post("/api/blueprints", body)

       def patch_blueprint(
           self,
           bp_id: str,
           name: str | None = None,
           status: str | None = None,
           version: str | None = None,
           summary: str | None = None,
       ) -> dict:
           body: dict[str, Any] = {}
           if name is not None:
               body["name"] = name
           if status is not None:
               body["status"] = status
           if version is not None:
               body["version"] = version
           if summary is not None:
               body["summary"] = summary
           return self._patch(f"/api/blueprints/{bp_id}", body)

       def archive_blueprint(self, bp_id: str) -> dict:
           return self._post(f"/api/blueprints/{bp_id}/archive")

       def export_blueprint(self, bp_id: str) -> str:
           """Returns raw markdown text."""
           return self._get_text(f"/api/blueprints/{bp_id}/export")

       def import_blueprint(self, markdown: str) -> dict:
           return self._post("/api/blueprints/import", {"markdown": markdown})

       # ------------------------------------------------------------------
       # nodes + edges
       # ------------------------------------------------------------------

       def add_node(
           self,
           bp_id: str,
           node_type: str,
           title: str,
           cat: str | None = None,
           node_id: str | None = None,
           body: str | None = None,
       ) -> dict:
           payload: dict[str, Any] = {"type": node_type, "title": title}
           if cat is not None:
               payload["cat"] = cat
           if node_id is not None:
               payload["id"] = node_id
           if body is not None:
               payload["body"] = body
           return self._post(f"/api/blueprints/{bp_id}/nodes", payload)

       def add_edge(
           self,
           bp_id: str,
           src: str,
           dst: str,
           label: str | None = None,
       ) -> dict:
           payload: dict[str, Any] = {"src": src, "dst": dst}
           if label is not None:
               payload["label"] = label
           return self._post(f"/api/blueprints/{bp_id}/edges", payload)

       # ------------------------------------------------------------------
       # assumptions / decisions / risks
       # ------------------------------------------------------------------

       def add_assumption(
           self,
           bp_id: str,
           assumption_id: str,
           bp_ref: str | None = None,
           status: str | None = None,
           risk: str | None = None,
           body: str | None = None,
       ) -> dict:
           payload: dict[str, Any] = {"id": assumption_id}
           if bp_ref is not None:
               payload["bp_ref"] = bp_ref
           if status is not None:
               payload["status"] = status
           if risk is not None:
               payload["risk"] = risk
           if body is not None:
               payload["body"] = body
           return self._post(f"/api/blueprints/{bp_id}/assumptions", payload)

       def add_decision(
           self,
           bp_id: str,
           decision_id: str,
           bp_ref: str | None = None,
           status: str | None = None,
           date: str | None = None,
           body: str | None = None,
       ) -> dict:
           payload: dict[str, Any] = {"id": decision_id}
           if bp_ref is not None:
               payload["bp_ref"] = bp_ref
           if status is not None:
               payload["status"] = status
           if date is not None:
               payload["date"] = date
           if body is not None:
               payload["body"] = body
           return self._post(f"/api/blueprints/{bp_id}/decisions", payload)

       def add_risk(
           self,
           bp_id: str,
           risk_id: str,
           risk: str | None = None,
           likelihood: str | None = None,
           impact: str | None = None,
           mitigation: str | None = None,
           status: str | None = None,
       ) -> dict:
           payload: dict[str, Any] = {"id": risk_id}
           if risk is not None:
               payload["risk"] = risk
           if likelihood is not None:
               payload["likelihood"] = likelihood
           if impact is not None:
               payload["impact"] = impact
           if mitigation is not None:
               payload["mitigation"] = mitigation
           if status is not None:
               payload["status"] = status
           return self._post(f"/api/blueprints/{bp_id}/risks", payload)

       # ------------------------------------------------------------------
       # chronicles
       # ------------------------------------------------------------------

       def add_chronicle(
           self,
           bp_id: str,
           kind: str = "raw",
           trigger: str | None = None,
           cycle: str | None = None,
           body: str | None = None,
       ) -> dict:
           payload: dict[str, Any] = {"kind": kind}
           if trigger is not None:
               payload["trigger"] = trigger
           if cycle is not None:
               payload["cycle"] = cycle
           if body is not None:
               payload["body"] = body
           return self._post(f"/api/blueprints/{bp_id}/chronicles", payload)

       def list_chronicles(self, bp_id: str) -> list:
           return self._get(f"/api/blueprints/{bp_id}/chronicles")

       # ------------------------------------------------------------------
       # rebuild
       # ------------------------------------------------------------------

       def rebuild(self) -> dict:
           return self._post("/api/rebuild")

       # ------------------------------------------------------------------
       # cleanup
       # ------------------------------------------------------------------

       def close(self) -> None:
           self._client.close()

       def __enter__(self) -> "CelestialClient":
           return self

       def __exit__(self, *_: Any) -> None:
           self.close()
   ```
