## Context

The CLI tests exercise `CelestialClient` against the real FastAPI `app` object using `httpx.ASGITransport` — the same approach used in T1.4.08 for the API tests, just routed through `CelestialClient` instead of `TestClient` directly. The fixture uses `monkeypatch.setenv` + `importlib.reload` to give each test run an isolated temp DB and blueprint/chronicle dirs. `CelestialClient` is constructed with a custom `httpx.Client` that uses `ASGITransport` instead of a real TCP connection, so no server process needs to be running.

## Objective

Write `tests/test_cli.py` covering blueprint CRUD, node add, chronicle add/list, export, and rebuild via `CelestialClient`.

## Instructions

1. Create `backend/tests/test_cli.py` with exactly:
   ```python
   """CLI integration tests: CelestialClient against in-process FastAPI via ASGITransport."""
   from __future__ import annotations

   import importlib
   import os
   import glob

   import httpx
   import pytest

   from cli.client import CelestialClient


   @pytest.fixture()
   def cli(tmp_path, monkeypatch):
       """Yield a CelestialClient wired to an isolated in-process FastAPI app."""
       monkeypatch.setenv("CELESTIAL_DB_PATH", str(tmp_path / "cli.db"))
       monkeypatch.setenv("CELESTIAL_BLUEPRINTS_DIR", str(tmp_path / "bps"))
       monkeypatch.setenv("CELESTIAL_CHRONICLES_DIR", str(tmp_path / "chron"))

       import app.core.config as config_mod
       importlib.reload(config_mod)
       import app.core.db as db_mod
       importlib.reload(db_mod)
       import app.services.store as store_mod
       importlib.reload(store_mod)
       import app.services.files as files_mod
       importlib.reload(files_mod)
       import app.main as main_mod
       importlib.reload(main_mod)

       transport = httpx.ASGITransport(app=main_mod.app)
       raw = httpx.Client(transport=transport, base_url="http://testserver")

       # Patch the internal httpx.Client on CelestialClient so it routes in-process
       client = CelestialClient.__new__(CelestialClient)
       client._client = raw
       yield client
       raw.close()


   # ---------------------------------------------------------------------------
   # health
   # ---------------------------------------------------------------------------

   def test_health(cli):
       result = cli.health()
       assert result["status"] == "ok"


   # ---------------------------------------------------------------------------
   # blueprint CRUD
   # ---------------------------------------------------------------------------

   def test_create_and_get(cli):
       bp = cli.create_blueprint(name="Test Plan", tier="small")
       bp_id = bp["frontmatter"]["id"]
       assert bp["frontmatter"]["name"] == "Test Plan"

       fetched = cli.get_blueprint(bp_id)
       assert fetched["frontmatter"]["id"] == bp_id


   def test_list_blueprints(cli):
       cli.create_blueprint(name="Alpha")
       cli.create_blueprint(name="Beta")
       listing = cli.list_blueprints()
       assert len(listing) >= 2
       names = {item["name"] for item in listing}
       assert "Alpha" in names and "Beta" in names


   def test_patch_blueprint(cli):
       bp = cli.create_blueprint(name="Patchable")
       bp_id = bp["frontmatter"]["id"]
       updated = cli.patch_blueprint(bp_id, status="active")
       assert updated["frontmatter"]["status"] == "active"


   def test_archive_blueprint(cli):
       bp = cli.create_blueprint(name="ToArchive")
       bp_id = bp["frontmatter"]["id"]
       result = cli.archive_blueprint(bp_id)
       assert result["status"] == "archived"
       assert result["id"] == bp_id


   # ---------------------------------------------------------------------------
   # node add
   # ---------------------------------------------------------------------------

   def test_add_node(cli):
       bp = cli.create_blueprint(name="NodePlan", tier="medium")
       bp_id = bp["frontmatter"]["id"]
       updated = cli.add_node(bp_id, node_type="text", title="Node One", node_id="n1")
       assert len(updated["nodes"]) == 1
       assert updated["nodes"][0]["title"] == "Node One"


   # ---------------------------------------------------------------------------
   # chronicle add + list
   # ---------------------------------------------------------------------------

   def test_chronicle_add_and_list(cli):
       bp = cli.create_blueprint(name="ChronPlan")
       bp_id = bp["frontmatter"]["id"]

       result = cli.add_chronicle(bp_id, kind="raw", body="First note")
       assert "path" in result
       assert os.path.exists(result["path"])

       entries = cli.list_chronicles(bp_id)
       assert len(entries) == 1


   # ---------------------------------------------------------------------------
   # export
   # ---------------------------------------------------------------------------

   def test_export_blueprint(cli):
       bp = cli.create_blueprint(name="ExportMe", tier="small")
       bp_id = bp["frontmatter"]["id"]
       md = cli.export_blueprint(bp_id)
       assert isinstance(md, str)
       assert md.startswith("---")
       assert "ExportMe" in md


   # ---------------------------------------------------------------------------
   # rebuild recovery
   # ---------------------------------------------------------------------------

   def test_rebuild(cli, tmp_path):
       bp = cli.create_blueprint(name="RecoverMe")
       bp_id = bp["frontmatter"]["id"]

       # Wipe the DB files to simulate loss
       for f in glob.glob(str(tmp_path / "cli.db*")):
           os.remove(f)

       result = cli.rebuild()
       assert result["imported"] >= 1

       # Blueprint should be recoverable
       fetched = cli.get_blueprint(bp_id)
       assert fetched["frontmatter"]["id"] == bp_id
   ```
