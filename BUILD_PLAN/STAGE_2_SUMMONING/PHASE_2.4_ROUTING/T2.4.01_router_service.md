## Context

The routing layer selects the best available agent for a given task type. It queries the DB
for `active` agents with the required capability, sorts by a priority order
(`claude-code` > `opencode` > `antigravity` > `codex` > `hermes` > `ollama` > custom), and
returns the top match. If no active agent has the capability, it falls back to `claude-code`
regardless of status (it is always assumed available on this system).

## Objective

Create `backend/app/services/router.py` with `route_task(task_type, context)` returning a
routing decision dict.

## Instructions

Create `backend/app/services/router.py` with this content:

```python
"""
Celestial Blueprint — Task Router
Selects the best agent for a given task_type from the registry.
"""
from __future__ import annotations

from app.services.agent_store import list_agents

# Priority order — lower index = higher priority
_PRIORITY: list[str] = [
    "claude-code",
    "opencode",
    "antigravity",
    "codex",
    "hermes",
    "ollama",
]

_FALLBACK_AGENT_ID = "claude-code"
_FALLBACK_CMD = "claude"


def route_task(task_type: str, context: str = "") -> dict:
    """
    Return a routing decision for the given task_type.

    Returns:
        {
            "agent_id": str,
            "agent_name": str,
            "activation_cmd": str,
            "task_type": str,
            "reason": str,
        }
    """
    # Fetch all active agents from DB
    active_agents = list_agents(status="active")

    # Filter to those with the requested capability
    capable = [
        a for a in active_agents
        if task_type in a.capabilities
    ]

    if not capable:
        # No active agent has the capability — fall back
        return _fallback_response(task_type, f"no active agent with capability '{task_type}'")

    # Sort by priority
    def priority_key(agent):
        try:
            return _PRIORITY.index(agent.id)
        except ValueError:
            return len(_PRIORITY)  # custom agents sort last

    capable.sort(key=priority_key)
    chosen = capable[0]

    return {
        "agent_id": chosen.id,
        "agent_name": chosen.name,
        "activation_cmd": chosen.activation_cmd,
        "task_type": task_type,
        "reason": f"highest-priority active agent with capability '{task_type}'",
    }


def _fallback_response(task_type: str, reason: str) -> dict:
    return {
        "agent_id": _FALLBACK_AGENT_ID,
        "agent_name": "Claude Code (fallback)",
        "activation_cmd": _FALLBACK_CMD,
        "task_type": task_type,
        "reason": f"fallback — {reason}",
    }
```
