## Context

The React frontend lives in `frontend/src/`. It uses TypeScript, React 18, React Router 6,
and Tailwind CSS. **Match the Stage 1 conventions exactly:**

- Tailwind uses named **theme-token classes**, not arbitrary hex. `tailwind.config.js`
  (T1.1.06) defines `bg` (#0b0e14), `surface` (#11151f), `border` (#1e2430), `ink` (#e6e9ef),
  `muted` (#8b93a7), `accent` (#7c5cff). Use `bg-bg`, `bg-surface`, `border-border`,
  `text-ink`, `text-muted`, `bg-accent`, `hover:bg-violet-500` — exactly like
  `BlueprintList.tsx` (T1.6.03). Do NOT use `bg-[#0b0e14]`-style arbitrary values.
- `frontend/src/lib/api.ts` (T1.6.01/T1.6.02) exports **named functions** (e.g.
  `export const listBlueprints = () => req<...>("GET", "/blueprints")`). `req` is
  module-private; `BASE` is already `"/api"`, so endpoint paths **omit** the `/api` prefix.
  There is NO `agents` namespace object — add named functions matching the existing style.
- Components import API **functions** from `../lib/api` and **types** from `../types/blueprint`.

The Agents page fetches the agents list, displays a card per agent with name, status badge,
capability chips, and a global Rescan button.

### 3. Create `frontend/src/pages/Agents.tsx`

```tsx
import { useEffect, useState } from "react";
import { listAgents, scanAgents } from "../lib/api";
import type { Agent } from "../types/blueprint";

const STATUS_COLOR: Record<string, string> = {
  active: "text-emerald-400",
  inactive: "text-amber-400",
  unknown: "text-muted",
};

export default function Agents() {
  const [agentList, setAgentList] = useState<Agent[]>([]);
  const [loading, setLoading] = useState(true);
  const [scanning, setScanning] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    listAgents()
      .then(setAgentList)
      .catch((e) => setError(String(e)))
      .finally(() => setLoading(false));
  }, []);

  async function handleRescan() {
    setScanning(true);
    setError(null);
    try {
      setAgentList(await scanAgents());
    } catch (e) {
      setError(String(e));
    } finally {
      setScanning(false);
    }
  }

  return (
    <div className="mx-auto max-w-4xl px-4 py-8">
      <div className="mb-6 flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-semibold text-ink">Agent Registry</h1>
          <p className="mt-1 text-sm text-muted">
            {agentList.length} agent{agentList.length !== 1 ? "s" : ""} discovered
          </p>
        </div>
        <button
          onClick={handleRescan}
          disabled={scanning}
          className="rounded bg-accent px-4 py-1.5 text-sm font-medium text-white hover:bg-violet-500 disabled:opacity-50"
        >
          {scanning ? "Scanning…" : "Rescan"}
        </button>
      </div>

      {error && (
        <div className="mb-6 rounded-lg border border-red-700 bg-surface p-4 text-sm text-red-400">
          {error}
        </div>
      )}

      {loading ? (
        <p className="text-sm text-muted">Loading…</p>
      ) : agentList.length === 0 ? (
        <div className="py-20 text-center text-muted">
          <p className="mb-2 text-lg">No agents registered.</p>
          <p className="text-sm">Click Rescan to discover installed agents.</p>
        </div>
      ) : (
        <ul className="space-y-3">
          {agentList.map((agent) => (
            <li
              key={agent.id}
              className="rounded-lg border border-border bg-surface p-5"
            >
              <div className="mb-1 flex items-center gap-3">
                <h2 className="truncate font-medium text-ink">{agent.name}</h2>
                <span
                  className={`text-xs font-medium ${STATUS_COLOR[agent.status] ?? "text-muted"}`}
                >
                  {agent.status}
                </span>
              </div>
              <p className="mb-3 font-mono text-xs text-muted">
                {agent.activation_cmd}
                {agent.version && <span className="ml-2 opacity-60">v{agent.version}</span>}
              </p>
              <div className="flex flex-wrap gap-1.5">
                {agent.capabilities.map((cap) => (
                  <span
                    key={cap}
                    className="rounded border border-border bg-bg px-2 py-0.5 text-xs text-muted"
                  >
                    {cap}
                  </span>
                ))}
              </div>
              {agent.models.length > 0 && (
                <p className="mt-2 text-xs text-muted">
                  Models: {agent.models.join(", ")}
                </p>
              )}
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```
