```python
def detect_hermes() -> Optional[dict]:
    """Detect Hermes Agent (task runner / automation framework)."""
    binary = detect_binary("hermes")
    # Also check known local install paths
    local_paths = [
        "~/.hermes/hermes-agent",
        "~/.hermes",
    ]
    found_local = any(detect_path(p) for p in local_paths)

    if not binary and not found_local:
        return None

    activation_cmd = binary or "hermes"

    version = ""
    if binary:
        try:
            import subprocess
            result = subprocess.run(
                ["hermes", "--version"], capture_output=True, text=True, timeout=5
            )
            version = result.stdout.strip().split("\n")[0]
        except Exception:
            pass

    status = "active" if binary else "inactive"

    return agent_entry(
        id="hermes",
        name="Hermes Agent",
        capabilities=["plan", "file_ops", "code_gen", "summarise"],
        activation_cmd=activation_cmd,
        status=status,
        version=version,
    )
```
