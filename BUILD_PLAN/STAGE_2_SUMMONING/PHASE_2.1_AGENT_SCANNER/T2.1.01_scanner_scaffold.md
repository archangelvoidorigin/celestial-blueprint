"""
Celestial Blueprint — Agent Scanner
Detects installed AI agent runtimes on the host system.
Each detector returns an entry dict or None.
scan_all() aggregates all detectors and writes discovered.json.
"""
from __future__ import annotations

import json
import os
import shutil
from pathlib import Path
from typing import Optional

# ── helpers ──────────────────────────────────────────────────────────────────

def detect_binary(name: str) -> Optional[str]:
    """Return full path to binary if it exists on PATH, else None."""
    return shutil.which(name)


def detect_path(path: str | Path) -> bool:
    """Return True if the filesystem path exists."""
    return Path(os.path.expanduser(str(path))).exists()


def agent_entry(
    *,
    id: str,
    name: str,
    capabilities: list[str],
    activation_cmd: str,
    status: str = "unknown",
    version: str = "",
) -> dict:
    """Build a normalised agent manifest entry."""
    return {
        "id": id,
        "name": name,
        "capabilities": capabilities,
        "activation_cmd": activation_cmd,
        "status": status,
        "version": version,
    }
