```python
def detect_custom_agents() -> list[dict]:
    """
    Detect any agents registered as directories under ~/.agents/.
    Each sub-directory may contain a manifest.json with keys:
    id, name, capabilities, activation_cmd, version.
    """
    agents_dir = Path.home() / ".agents"
    if not agents_dir.exists():
        return []

    results = []
    for entry in agents_dir.iterdir():
        manifest_file = entry / "manifest.json"
        if entry.is_dir() and manifest_file.exists():
            try:
                with open(manifest_file) as f:
                    data = json.load(f)
                results.append(
                    agent_entry(
                        id=data.get("id", entry.name),
                        name=data.get("name", entry.name),
                        capabilities=data.get("capabilities", []),
                        activation_cmd=data.get("activation_cmd", entry.name),
                        status="active",
                        version=data.get("version", ""),
                    )
                )
            except Exception:
                pass  # malformed manifest — skip
    return results


# ── aggregator ───────────────────────────────────────────────────────────────

def scan_all() -> list[dict]:
    """Run all detectors and return deduplicated list of found agents."""
    single_detectors = [
        detect_claude_code,
        detect_opencode,
        detect_codex,
        detect_antigravity,
        detect_hermes,
        detect_ollama,
    ]
    found: list[dict] = []
    for fn in single_detectors:
        result = fn()
        if result is not None:
            found.append(result)

    found.extend(detect_custom_agents())
    return found


def write_discovered_json(agents: list[dict], path: Path | None = None) -> Path:
    """Write the agents list to discovered.json. Returns the written path."""
    if path is None:
        path = Path(__file__).parent / "discovered.json"
    path.parent.mkdir(parents=True, exist_ok=True)
    with open(path, "w") as f:
        json.dump(agents, f, indent=2)
    return path


# ── CLI entry point ───────────────────────────────────────────────────────────

if __name__ == "__main__":
    agents = scan_all()
    out = write_discovered_json(agents)
    print(f"Discovered {len(agents)} agent(s) → {out}")
    for a in agents:
        print(f"  [{a['status']}] {a['id']}  {a.get('version','')}")
```
