```python
def detect_codex() -> Optional[dict]:
    """Detect OpenAI Codex CLI."""
    binary = detect_binary("codex")
    if not binary:
        return None

    version = ""
    try:
        import subprocess
        result = subprocess.run(
            ["codex", "--version"], capture_output=True, text=True, timeout=5
        )
        version = result.stdout.strip().split("\n")[0]
    except Exception:
        pass

    # Status depends on whether API key is set
    has_key = bool(os.environ.get("OPENAI_API_KEY"))
    status = "active" if has_key else "inactive"

    return agent_entry(
        id="codex",
        name="OpenAI Codex CLI",
        capabilities=["code_gen", "file_ops", "web_search"],
        activation_cmd=binary,
        status=status,
        version=version,
    )
```
