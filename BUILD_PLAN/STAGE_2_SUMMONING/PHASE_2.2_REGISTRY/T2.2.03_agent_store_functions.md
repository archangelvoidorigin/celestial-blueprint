## Context

The agent registry needs a service layer to read/write agents in SQLite. The existing
`backend/app/services/store.py` handles blueprints. By convention each resource group gets
its own store module. This task creates `agent_store.py` with five functions:
`upsert_agent`, `list_agents`, `get_agent`, `patch_agent`, and `delete_agent`. Capabilities
and models are stored as JSON strings (sqlite TEXT column).

Match the Stage 1 `store.py` data-access pattern exactly: import the `connection()` context
manager from `app.core.db` (it yields a `sqlite3.Row`-factory connection, commits on success,
and closes on exit — so writes need no explicit `conn.commit()`/`conn.close()`). Timestamps
use the same helper Stage 1 uses: `datetime.now().isoformat(timespec="seconds")`. There is no
`get_connection` in `db.py` — the functions are `get_conn`, `init_db`, `init_agents`, and
`connection`.

## Objective

Create `backend/app/services/agent_store.py` with full CRUD for the agents table.

Create `backend/app/services/agent_store.py` with this content:

```python
"""
Celestial Blueprint — Agent Store
CRUD operations for the agents registry table.
"""
from __future__ import annotations

import json
from datetime import datetime

from app.core.db import connection
from app.models.schemas import Agent, AgentCreate, AgentPatch


def _now() -> str:
    return datetime.now().isoformat(timespec="seconds")


def _row_to_agent(row) -> Agent:
    return Agent(
        id=row["id"],
        name=row["name"],
        capabilities=json.loads(row["capabilities"]),
        activation_cmd=row["activation_cmd"] or "",
        status=row["status"],
        version=row["version"] or "",
        models=json.loads(row["models"]),
        discovered_at=row["discovered_at"],
        updated_at=row["updated_at"],
    )


def upsert_agent(data: AgentCreate) -> Agent:
    """Insert or replace an agent entry (keyed on id)."""
    now = _now()
    with connection() as c:
        c.execute(
            """
            INSERT INTO agents (id, name, capabilities, activation_cmd, status, version,
                                models, discovered_at, updated_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            ON CONFLICT(id) DO UPDATE SET
                name = excluded.name,
                capabilities = excluded.capabilities,
                activation_cmd = excluded.activation_cmd,
                status = excluded.status,
                version = excluded.version,
                models = excluded.models,
                updated_at = excluded.updated_at
            """,
            (
                data.id,
                data.name,
                json.dumps(data.capabilities),
                data.activation_cmd,
                data.status,
                data.version,
                json.dumps(data.models),
                now,
                now,
            ),
        )
        row = c.execute("SELECT * FROM agents WHERE id = ?", (data.id,)).fetchone()
    return _row_to_agent(row)


def list_agents(status: str | None = None) -> list[Agent]:
    """List all agents, optionally filtered by status."""
    with connection() as c:
        if status:
            rows = c.execute(
                "SELECT * FROM agents WHERE status = ? ORDER BY name", (status,)
            ).fetchall()
        else:
            rows = c.execute("SELECT * FROM agents ORDER BY name").fetchall()
    return [_row_to_agent(r) for r in rows]


def get_agent(agent_id: str) -> Agent | None:
    """Get a single agent by id, or None if not found."""
    with connection() as c:
        row = c.execute("SELECT * FROM agents WHERE id = ?", (agent_id,)).fetchone()
    return _row_to_agent(row) if row else None


def patch_agent(agent_id: str, patch: AgentPatch) -> Agent | None:
    """Partially update an agent. Returns updated agent or None if not found."""
    existing = get_agent(agent_id)
    if not existing:
        return None

    updates: dict = patch.model_dump(exclude_none=True)
    if not updates:
        return existing

    updates["updated_at"] = _now()

    # JSON-encode list fields
    for field in ("capabilities", "models"):
        if field in updates:
            updates[field] = json.dumps(updates[field])

    set_clause = ", ".join(f"{k} = ?" for k in updates)
    values = list(updates.values()) + [agent_id]

    with connection() as c:
        c.execute(f"UPDATE agents SET {set_clause} WHERE id = ?", values)
        row = c.execute("SELECT * FROM agents WHERE id = ?", (agent_id,)).fetchone()
    return _row_to_agent(row)


def delete_agent(agent_id: str) -> bool:
    """Delete an agent. Returns True if a row was deleted."""
    with connection() as c:
        cursor = c.execute("DELETE FROM agents WHERE id = ?", (agent_id,))
        deleted = cursor.rowcount > 0
    return deleted
```
